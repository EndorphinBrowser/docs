48 QStringList AcceptLanguageDialog::expand(const QLocale::Language language)
49 {
50  QStringList allLanguages;
51  QList<QLocale::Country> countries = QLocale::countriesForLanguage(language);
52  for (int j = 0; j < countries.size(); ++j) {
53  QString languageString;
54  if (countries.count() == 1) {
55  languageString = QString(QLatin1String("%1 [%2]"))
56  .arg(QLocale::languageToString(language))
57  .arg(QLocale(language).name().split(QLatin1Char('_')).at(0));
58  } else {
59  languageString = QString(QLatin1String("%1/%2 [%3]"))
60  .arg(QLocale::languageToString(language))
61  .arg(QLocale::countryToString(countries.at(j)))
62  .arg(QLocale(language, countries.at(j)).name().split(QLatin1Char('_')).join(QLatin1String("-")).toLower());
63 
64  }
65  if (!allLanguages.contains(languageString))
66  allLanguages.append(languageString);
67  }
68  return allLanguages;
69 }

112 /*
113  Return a ByteArray that can be sent along with the Accept-Language http header
114 
115  See RFC 2616 section 14.4
116  */
117 QByteArray AcceptLanguageDialog::httpString(const QStringList &list)
118 {
119  QStringList processed;
120  qreal qvalue = 1.0;
121  foreach (const QString &string, list) {
122  int leftBracket = string.indexOf(QLatin1Char('['));
123  int rightBracket = string.indexOf(QLatin1Char(']'));
124  QString tag = string.mid(leftBracket + 1, rightBracket - leftBracket - 1);
125  if (processed.isEmpty()) {
126  processed << tag;
127  } else {
128  processed << QString(QLatin1String("%1; %2")).arg(tag).arg(QString::number(qvalue, 'f', 1));
129  }
130  if (qvalue > .1)
131  qvalue -= .1;
132  }
133  return processed.join(QLatin1String(", ")).toLatin1();
134 }
