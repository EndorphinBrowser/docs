204 void AddBookmarkDialog::accept()
205 {
206  if ((!m_addFolder && address->text().isEmpty()) || name->text().isEmpty()) {
207  QDialog::accept();
208  return;
209  }
210 
211  QModelIndex index = currentIndex();
212  if (!index.isValid())
213  index = m_bookmarksManager->bookmarksModel()->index(0, 0);
214  BookmarkNode *parent = m_bookmarksManager->bookmarksModel()->node(index);
215 
216  BookmarkNode::Type type = (m_addFolder) ? BookmarkNode::Folder : BookmarkNode::Bookmark;
217  BookmarkNode *bookmark = new BookmarkNode(type);
218  bookmark->title = name->text();
219  if (!m_addFolder)
220  bookmark->url = address->text();
221 
222  m_bookmarksManager->addBookmark(parent, bookmark);
223  m_addedNode = bookmark;
224 
225  QDialog::accept();
226 }
