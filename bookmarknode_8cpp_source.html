120 void BookmarkNode::add(BookmarkNode *child, int offset)
121 {
122  if (!child)
123  return;
124  if (m_type == BookmarkNode::Bookmark)
125  return;
126  Q_ASSERT(child->m_type != Root);
127  if (child->m_parent)
128  child->m_parent->remove(child);
129  child->m_parent = this;
130  if (-1 == offset)
131  offset = m_children.size();
132  m_children.insert(offset, child);
133 }
