120 bool BookmarksDialog::saveExpandedNodes(const QModelIndex &parent)
121 {
122  bool changed = false;
123  for (int i = 0; i < m_proxyModel->rowCount(parent); ++i) {
124  QModelIndex child = m_proxyModel->index(i, 0, parent);
125  QModelIndex sourceIndex = m_proxyModel->mapToSource(child);
126  BookmarkNode *childNode = m_bookmarksModel->node(sourceIndex);
127  bool wasExpanded = childNode->expanded;
128  if (tree->isExpanded(child)) {
129  childNode->expanded = true;
130  changed |= saveExpandedNodes(child);
131  } else {
132  childNode->expanded = false;
133  }
134  changed |= (wasExpanded != childNode->expanded);
135  }
136  return changed;
137 }

152 void BookmarksDialog::customContextMenuRequested(const QPoint &pos)
153 {
154  QMenu menu;
155  QModelIndex index = tree->indexAt(pos);
156  index = index.sibling(index.row(), 0);
157  QModelIndex sourceIndex = m_proxyModel->mapToSource(index);
158  const BookmarkNode *node = m_bookmarksModel->node(sourceIndex);
159  if (index.isValid() && node->type() != BookmarkNode::Folder) {
160  menu.addAction(tr("Open"), this, SLOT(openInCurrentTab()));
161  menu.addAction(tr("Open in New Tab"), this, SLOT(openInNewTab()));
162  menu.addSeparator();
163  }
164  menu.addSeparator();
165  QAction *renameAction = menu.addAction(tr("Edit Name"), this, SLOT(editName()));
166  renameAction->setEnabled(index.flags() & Qt::ItemIsEditable);
167  if (index.isValid() && node->type() != BookmarkNode::Folder) {
168  menu.addAction(tr("Edit Address"), this, SLOT(editAddress()));
169  }
170  menu.addSeparator();
171  QAction *deleteAction = menu.addAction(tr("Delete"), tree, SLOT(removeSelected()));
172  deleteAction->setEnabled(index.flags() & Qt::ItemIsDragEnabled);
173  menu.exec(QCursor::pos());
174 }

176 void BookmarksDialog::openBookmark(TabWidget::OpenUrlIn tab)
177 {
178  QModelIndex index = tree->currentIndex();
179  QModelIndex sourceIndex = m_proxyModel->mapToSource(index);
180  const BookmarkNode *node = m_bookmarksModel->node(sourceIndex);
181  if (!index.parent().isValid() || !node || node->type() == BookmarkNode::Folder)
182  return;
183  emit openUrl(
184  index.sibling(index.row(), 1).data(BookmarksModel::UrlRole).toUrl(),
185  tab,
186  index.sibling(index.row(), 0).data(Qt::DisplayRole).toString());
187 }

220 void BookmarksDialog::newFolder()
221 {
222  QModelIndex currentIndex = tree->currentIndex();
223  QModelIndex idx = currentIndex;
224  QModelIndex sourceIndex = m_proxyModel->mapToSource(idx);
225  const BookmarkNode *sourceNode = m_bookmarksModel->node(sourceIndex);
226  int row = -1; // default: append new folder as last item in selected folder
227  if (sourceNode && sourceNode->type() != BookmarkNode::Folder) {
228  // if selected item is not a folder, add new folder to parent folder,
229  // but direcly below the selected item
230  idx = idx.parent();
231  row = currentIndex.row() + 1;
232  }
233  if (!idx.isValid())
234  idx = m_proxyModel->index(1, 0); // Select Bookmarks menu as default
235  idx = m_proxyModel->mapToSource(idx);
236  BookmarkNode *parent = m_bookmarksManager->bookmarksModel()->node(idx);
237  BookmarkNode *node = new BookmarkNode(BookmarkNode::Folder);
238  node->title = tr("New Folder");
239  m_bookmarksManager->addBookmark(parent, node, row);
240 }
