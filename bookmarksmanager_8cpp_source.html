120 void BookmarksManager::load()
121 {
122  if (m_loaded)
123  return;
124  m_loaded = true;
125 
126  QString dir = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation) + "/data/Endorphin";
127  QString bookmarkFile = dir + QLatin1String("/bookmarks.xbel");
128  if (!QFile::exists(bookmarkFile))
129  bookmarkFile = QLatin1String(":defaultbookmarks.xbel");
130 
131  XbelReader reader;
132  m_bookmarkRootNode = reader.read(bookmarkFile);
133  if (reader.error() != QXmlStreamReader::NoError) {
134  QMessageBox::warning(nullptr, QLatin1String("Loading Bookmark"),
135  tr("Error when loading bookmarks on line %1, column %2:\n"
136  "%3").arg(reader.lineNumber()).arg(reader.columnNumber()).arg(reader.errorString()));
137  }
138 
139  QList<BookmarkNode*> others;
140  for (int i = m_bookmarkRootNode->children().count() - 1; i >= 0; --i) {
141  BookmarkNode *node = m_bookmarkRootNode->children().at(i);
142  if (node->type() == BookmarkNode::Folder) {
143  // Automatically convert
144  if ((node->title == tr("Toolbar Bookmarks")
145  || node->title == QLatin1String(BOOKMARKBAR)) && !m_toolbar) {
146  node->title = tr(BOOKMARKBAR);
147 
148  m_toolbar = node;
149  }
150 
151  // Automatically convert
152  if ((node->title == tr("Menu")
153  || node->title == QLatin1String(BOOKMARKMENU)) && !m_menu) {
154  node->title = tr(BOOKMARKMENU);
155  m_menu = node;
156  }
157  } else {
158  others.append(node);
159  }
160  m_bookmarkRootNode->remove(node);
161  }
162  Q_ASSERT(m_bookmarkRootNode->children().count() == 0);
163  if (!m_toolbar) {
164  m_toolbar = new BookmarkNode(BookmarkNode::Folder, m_bookmarkRootNode);
165  m_toolbar->title = tr(BOOKMARKBAR);
166  } else {
167  m_bookmarkRootNode->add(m_toolbar);
168  }
169 
170  if (!m_menu) {
171  m_menu = new BookmarkNode(BookmarkNode::Folder, m_bookmarkRootNode);
172  m_menu->title = tr(BOOKMARKMENU);
173  } else {
174  m_bookmarkRootNode->add(m_menu);
175  }
176 
177  for (int i = 0; i < others.count(); ++i)
178  m_menu->add(others.at(i));
179 }

181 void BookmarksManager::save() const
182 {
183  if (!m_loaded)
184  return;
185 
186  XbelWriter writer;
187  QString dir = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation) + "/data/Endorphin";
188  QString bookmarkFile = dir + QLatin1String("/bookmarks.xbel");
189  // Save root folder titles in English (i.e. not localized)
190  m_menu->title = QLatin1String(BOOKMARKMENU);
191  m_toolbar->title = QLatin1String(BOOKMARKBAR);
192  if (!writer.write(bookmarkFile, m_bookmarkRootNode))
193  qWarning() << "BookmarkManager: error saving to" << bookmarkFile;
194  // Restore localized titles
195  retranslate();
196 }

279 void BookmarksManager::importBookmarks()
280 {
281  QStringList supportedFormats;
282  supportedFormats << tr("XBEL bookmarks").append(QLatin1String("(*.xbel *.xml)"));
283  supportedFormats << tr("HTML Netscape bookmarks").append(QLatin1String("(*.html)"));
284 
285  QString fileName = QFileDialog::getOpenFileName(nullptr, tr("Open File"),
286  QString(), supportedFormats.join(QLatin1String(";;")));
287  if (fileName.isEmpty())
288  return;
289 
290  XbelReader reader;
291  BookmarkNode *importRootNode = nullptr;
292  if (fileName.endsWith(QLatin1String(".html"))) {
293  QString program = QLatin1String("htmlToXBel");
294  QStringList arguments;
295  arguments << fileName;
296  QProcess process;
297  process.start(program, arguments);
298  process.waitForFinished(-1);
299  if (process.error() != QProcess::UnknownError) {
300  if (process.error() == QProcess::FailedToStart) {
301  QMessageBox::warning(nullptr, tr("htmlToXBel tool required"),
302  tr("htmlToXBel tool, which is shipped with Endorphin and is needed to import HTML bookmarks, "
303  "is not installed or not available in the search paths."));
304  } else {
305  QMessageBox::warning(nullptr, tr("Loading Bookmark"),
306  tr("Error when loading HTML bookmarks: %1\n").arg(process.errorString()));
307  }
308  return;
309  }
310  importRootNode = reader.read(&process);
311  } else {
312  importRootNode = reader.read(fileName);
313  }
314  if (reader.error() != QXmlStreamReader::NoError) {
315  QMessageBox::warning(nullptr, QLatin1String("Loading Bookmark"),
316  tr("Error when loading bookmarks on line %1, column %2:\n"
317  "%3").arg(reader.lineNumber()).arg(reader.columnNumber()).arg(reader.errorString()));
318  delete importRootNode;
319  return;
320  }
321 
322  importRootNode->setType(BookmarkNode::Folder);
323  importRootNode->title = (tr("Imported %1").arg(QDate::currentDate().toString(Qt::SystemLocaleShortDate)));
324  addBookmark(menu(), importRootNode);
325 }
