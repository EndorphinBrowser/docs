99 void BookmarksMenu::postPopulated()
100 {
101  if (isEmpty())
102  return;
103 
104  QModelIndex parent = rootIndex();
105 
106  bool hasBookmarks = false;
107 
108  for (int i = 0; i < parent.model()->rowCount(parent); ++i) {
109  QModelIndex child = parent.model()->index(i, 0, parent);
110 
111  if (child.data(BookmarksModel::TypeRole) == BookmarkNode::Bookmark) {
112  hasBookmarks = true;
113  break;
114  }
115  }
116 
117  if (!hasBookmarks)
118  return;
119 
120  addSeparator();
121  QAction *action = addAction(tr("Open in Tabs"));
122  connect(action, SIGNAL(triggered()),
123  this, SLOT(openAll()));
124 }

126 void BookmarksMenu::openAll()
127 {
128  ModelMenu *menu = qobject_cast<ModelMenu*>(sender()->parent());
129  if (!menu)
130  return;
131  QModelIndex parent = menu->rootIndex();
132  if (!parent.isValid())
133  return;
134  for (int i = 0; i < parent.model()->rowCount(parent); ++i) {
135  QModelIndex child = parent.model()->index(i, 0, parent);
136 
137  if (child.data(BookmarksModel::TypeRole) != BookmarkNode::Bookmark)
138  continue;
139 
140  TabWidget::OpenUrlIn tab;
141  tab = (i == 0) ? TabWidget::CurrentTab : TabWidget::NewTab;
142  emit openUrl(child.data(BookmarksModel::UrlRole).toUrl(),
143  tab,
144  child.data(Qt::DisplayRole).toString());
145  }
146 }
