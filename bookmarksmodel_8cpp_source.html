100 void BookmarksModel::entryAdded(BookmarkNode *item)
101 {
102  Q_ASSERT(item && item->parent());
103  int row = item->parent()->children().indexOf(item);
104  BookmarkNode *parent = item->parent();
105  // item was already added so remove beore beginInsertRows is called
106  parent->remove(item);
107  beginInsertRows(index(parent), row, row);
108  parent->add(item, row);
109  endInsertRows();
110 }

127 bool BookmarksModel::removeRows(int row, int count, const QModelIndex &parent)
128 {
129  if (row < 0 || count <= 0 || row + count > rowCount(parent))
130  return false;
131 
132  BookmarkNode *bookmarkNode = node(parent);
133  for (int i = row + count - 1; i >= row; --i) {
134  BookmarkNode *node = bookmarkNode->children().at(i);
135  if (node == m_bookmarksManager->menu()
136  || node == m_bookmarksManager->toolbar())
137  continue;
138 
139  m_bookmarksManager->removeBookmark(node);
140  }
141  if (m_endMacro) {
142  m_bookmarksManager->undoRedoStack()->endMacro();
143  m_endMacro = false;
144  }
145  return true;
146 }

161 QVariant BookmarksModel::data(const QModelIndex &index, int role) const
162 {
163  if (!index.isValid() || index.model() != this)
164  return QVariant();
165 
166  const BookmarkNode *bookmarkNode = node(index);
167  switch (role) {
168  case Qt::EditRole:
169  case Qt::DisplayRole:
170  if (bookmarkNode->type() == BookmarkNode::Separator) {
171  switch (index.column()) {
172  case 0:
173  return QString(50, 0xB7);
174  case 1:
175  return QString();
176  }
177  }
178 
179  switch (index.column()) {
180  case 0:
181  return bookmarkNode->title;
182  case 1:
183  return bookmarkNode->url;
184  }
185  break;
186  case BookmarksModel::UrlRole:
187  return QUrl(bookmarkNode->url);
188  break;
189  case BookmarksModel::UrlStringRole:
190  return bookmarkNode->url;
191  break;
192  case BookmarksModel::TypeRole:
193  return bookmarkNode->type();
194  break;
195  case BookmarksModel::SeparatorRole:
196  return (bookmarkNode->type() == BookmarkNode::Separator);
197  break;
198  case Qt::DecorationRole:
199  if (index.column() == 0) {
200  if (bookmarkNode->type() == BookmarkNode::Folder)
201  return QApplication::style()->standardIcon(QStyle::SP_DirIcon);
202 #ifndef NO_BROWSERAPPLICATION
203  return BrowserApplication::instance()->icon(bookmarkNode->url);
204 #endif
205  }
206  }
207 
208  return QVariant();
209 }

238 QModelIndex BookmarksModel::parent(const QModelIndex &index) const
239 {
240  if (!index.isValid())
241  return QModelIndex();
242 
243  BookmarkNode *itemNode = node(index);
244  BookmarkNode *parentNode = (itemNode ? itemNode->parent() : nullptr);
245  if (!parentNode || parentNode == m_bookmarksManager->bookmarks())
246  return QModelIndex();
247 
248  // get the parent's row
249  BookmarkNode *grandParentNode = parentNode->parent();
250  int parentRow = grandParentNode->children().indexOf(parentNode);
251  Q_ASSERT(parentRow >= 0);
252  return createIndex(parentRow, 0, parentNode);
253 }

263 Qt::ItemFlags BookmarksModel::flags(const QModelIndex &index) const
264 {
265  if (!index.isValid())
266  return Qt::NoItemFlags;
267 
268  BookmarkNode *node = this->node(index);
269  BookmarkNode::Type type = node->type();
270  Qt::ItemFlags flags = Qt::ItemIsSelectable | Qt::ItemIsEnabled;
271 
272  if (hasChildren(index))
273  flags |= Qt::ItemIsDropEnabled;
274 
275  if (node == m_bookmarksManager->menu()
276  || node == m_bookmarksManager->toolbar())
277  return flags;
278 
279  flags |= Qt::ItemIsDragEnabled;
280 
281  if ((index.column() == 0 && type != BookmarkNode::Separator)
282  || (index.column() == 1 && type == BookmarkNode::Bookmark))
283  flags |= Qt::ItemIsEditable;
284 
285  return flags;
286 }

303 QMimeData *BookmarksModel::mimeData(const QModelIndexList &indexes) const
304 {
305  QMimeData *mimeData = new QMimeData();
306  QByteArray data;
307  QDataStream stream(&data, QIODevice::WriteOnly);
308  QList<QUrl> urls;
309  foreach (const QModelIndex &index, indexes) {
310  if (index.column() != 0 || !index.isValid())
311  continue;
312  QByteArray encodedData;
313  QBuffer buffer(&encodedData);
314  buffer.open(QBuffer::ReadWrite);
315  XbelWriter writer;
316  const BookmarkNode *parentNode = node(index);
317  writer.write(&buffer, parentNode);
318  stream << encodedData;
319  urls.append(index.data(BookmarksModel::UrlRole).toUrl());
320  }
321  mimeData->setData(MIMETYPE, data);
322  mimeData->setUrls(urls);
323  return mimeData;
324 }

326 bool BookmarksModel::dropMimeData(const QMimeData *data,
327  Qt::DropAction action, int row, int column, const QModelIndex &parent)
328 {
329  if (action == Qt::IgnoreAction)
330  return true;
331 
332  if (column > 0)
333  return false;
334 
335  BookmarkNode *parentNode = node(parent);
336 
337  if (!data->hasFormat(MIMETYPE)) {
338  if (!data->hasUrls())
339  return false;
340 
341  BookmarkNode *node = new BookmarkNode(BookmarkNode::Bookmark, parentNode);
342  node->url = QString::fromUtf8(data->urls().at(0).toEncoded());
343 
344  if (data->hasText())
345  node->title = data->text();
346  else
347  node->title = node->url;
348 
349  m_bookmarksManager->addBookmark(parentNode, node, row);
350  return true;
351  }
352 
353  QByteArray ba = data->data(MIMETYPE);
354  QDataStream stream(&ba, QIODevice::ReadOnly);
355  if (stream.atEnd())
356  return false;
357 
358  QUndoStack *undoStack = m_bookmarksManager->undoRedoStack();
359  undoStack->beginMacro(QLatin1String("Move Bookmarks"));
360 
361  while (!stream.atEnd()) {
362  QByteArray encodedData;
363  stream >> encodedData;
364  QBuffer buffer(&encodedData);
365  buffer.open(QBuffer::ReadOnly);
366 
367  XbelReader reader;
368  BookmarkNode *rootNode = reader.read(&buffer);
369  QList<BookmarkNode*> children = rootNode->children();
370  for (int i = 0; i < children.count(); ++i) {
371  BookmarkNode *bookmarkNode = children.at(i);
372  rootNode->remove(bookmarkNode);
373  row = qMax(0, row);
374  m_bookmarksManager->addBookmark(parentNode, bookmarkNode, row);
375  m_endMacro = true;
376  }
377  delete rootNode;
378  }
379  return true;
380 }

382 bool BookmarksModel::setData(const QModelIndex &index, const QVariant &value, int role)
383 {
384  if (!index.isValid() || (flags(index) & Qt::ItemIsEditable) == 0)
385  return false;
386 
387  BookmarkNode *item = node(index);
388 
389  switch (role) {
390  case Qt::EditRole:
391  case Qt::DisplayRole:
392  if (index.column() == 0) {
393  m_bookmarksManager->setTitle(item, value.toString());
394  break;
395  }
396  if (index.column() == 1) {
397  m_bookmarksManager->setUrl(item, value.toString());
398  break;
399  }
400  return false;
401  case BookmarksModel::UrlRole:
402  m_bookmarksManager->setUrl(item, value.toUrl().toString());
403  break;
404  case BookmarksModel::UrlStringRole:
405  m_bookmarksManager->setUrl(item, value.toString());
406  break;
407  default:
408  break;
409  return false;
410  }
411 
412  return true;
413 }
