65 void BookmarksToolBar::contextMenuRequested(const QPoint &position)
66 {
67  QAction *action = actionAt(position);
68 
69  QMenu menu;
70 
71  if (action) {
72  QVariant variant = action->data();
73  Q_ASSERT(variant.canConvert<QModelIndex>());
74 
75  QAction *menuAction = nullptr;
76 
77  if (!action->menu()) {
78  menuAction = menu.addAction(tr("Open"), this, SLOT(openBookmarkInCurrentTab()));
79  menuAction->setData(variant);
80 
81  menuAction = menu.addAction(tr("Open in New &Tab"), this, SLOT(openBookmarkInNewTab()));
82  menuAction->setData(variant);
83 
84  menu.addSeparator();
85  }
86 
87  menuAction = menu.addAction(tr("Remove"), this, SLOT(removeBookmark()));
88  menuAction->setData(variant);
89 
90  menu.addSeparator();
91  }
92 
93  menu.addAction(tr("Add Bookmark..."), this, SLOT(newBookmark()));
94  menu.addAction(tr("Add Folder..."), this, SLOT(newFolder()));
95 
96  menu.exec(QCursor::pos());
97 }
