148 static void setUserStyleSheet(QWebEngineProfile *profile, const QString &styleSheet, BrowserMainWindow *mainWindow = nullptr)
149 {
150  Q_ASSERT(profile);
151  QString scriptName(QStringLiteral("userStyleSheet"));
152  QWebEngineScript script;
153  QList<QWebEngineScript> styleSheets = profile->scripts()->findScripts(scriptName);
154  if (!styleSheets.isEmpty())
155  script = styleSheets.first();
156  Q_FOREACH (const QWebEngineScript &s, styleSheets)
157  profile->scripts()->remove(s);
158 
159  if (script.isNull()) {
160  script.setName(scriptName);
161  script.setInjectionPoint(QWebEngineScript::DocumentReady);
162  script.setRunsOnSubFrames(true);
163  script.setWorldId(QWebEngineScript::ApplicationWorld);
164  }
165  QString source = QString::fromLatin1("(function() {"\
166  "var css = document.getElementById(\"_qt_testBrowser_userStyleSheet\");"\
167  "if (css == undefined) {"\
168  " css = document.createElement(\"style\");"\
169  " css.type = \"text/css\";"\
170  " css.id = \"_qt_testBrowser_userStyleSheet\";"\
171  " document.head.appendChild(css);"\
172  "}"\
173  "css.innerText = \"%1\";"\
174  "})()").arg(styleSheet);
175  script.setSourceCode(source);
176  profile->scripts()->insert(script);
177  // run the script on the already loaded views
178  // this has to be deferred as it could mess with the storage initialization on startup
179  if (mainWindow)
180  QMetaObject::invokeMethod(mainWindow, "runScriptOnOpenViews", Qt::QueuedConnection, Q_ARG(QString, source));
181 }

183 BrowserApplication::BrowserApplication(int &argc, char **argv)
184  : SingleApplication(argc, argv)
185  , quitting(false)
186 {
187  QCoreApplication::setOrganizationDomain(QLatin1String("EndorphinBrowser.github.io/"));
188  QCoreApplication::setApplicationName(QLatin1String("Endorphin"));
189  QCoreApplication::setApplicationVersion(QLatin1String("0.12.1"
190 #ifdef GITVERSION
191  " (Git: " GITCHANGENUMBER " " GITVERSION ")"
192 #endif
193  ));
194 
195 #ifndef AUTOTESTS
196  connect(this, SIGNAL(messageReceived(QLocalSocket *)),
197  this, SLOT(messageReceived(QLocalSocket *)));
198 
199  QStringList args = QCoreApplication::arguments();
200  if (args.count() > 1) {
201  QString message = parseArgumentUrl(args.last());
202  sendMessage(message.toUtf8());
203  }
204  // If we could connect to another Endorphin then exit
205  QString message = QString(QLatin1String("endorphin://getwinid"));
206  if (sendMessage(message.toUtf8(), 500))
207  return;
208 
209 #ifdef BROWSERAPPLICATION_DEBUG
210  qDebug() << "BrowserApplication::" << __FUNCTION__ << "I am the only endorphin";
211 #endif
212 
213  // not sure what else to do...
214  if (!startSingleServer())
215  return;
216 #endif
217 
218 #if defined(Q_WS_MAC)
219  QApplication::setQuitOnLastWindowClosed(false);
220 #else
221  QApplication::setQuitOnLastWindowClosed(true);
222 #endif
223 
224  QDesktopServices::setUrlHandler(QLatin1String("http"), this, "openUrl");
225 
226  // Until QtWebkit defaults to 16
227  QWebEngineSettings::globalSettings()->setFontSize(QWebEngineSettings::DefaultFontSize, 16);
228  QWebEngineSettings::globalSettings()->setFontSize(QWebEngineSettings::DefaultFixedFontSize, 16);
229 
230  QSettings settings;
231  settings.beginGroup(QLatin1String("sessions"));
232  m_lastSession = settings.value(QLatin1String("lastSession")).toByteArray();
233  settings.endGroup();
234 
235 #if defined(Q_WS_MAC)
236  connect(this, SIGNAL(lastWindowClosed()),
237  this, SLOT(lastWindowClosed()));
238 #endif
239 
240  // setting this in the postLaunch actually takes a lot more time
241  // because the event has to be propagated to everyone.
242  setWindowIcon(QIcon(QLatin1String(":128x128/endorphin.png")));
243 
244 #ifndef AUTOTESTS
245  QTimer::singleShot(0, this, SLOT(postLaunch()));
246 #endif
247  languageManager();
248 }

291 void BrowserApplication::messageReceived(QLocalSocket *socket)
292 {
293  QString message;
294  QTextStream stream(socket);
295  stream >> message;
296 #ifdef BROWSERAPPLICATION_DEBUG
297  qDebug() << "BrowserApplication::" << __FUNCTION__ << message;
298 #endif
299  if (message.isEmpty())
300  return;
301 
302  // Got a normal url
303  if (!message.startsWith(QLatin1String("endorphin://"))) {
304  QSettings settings;
305  settings.beginGroup(QLatin1String("tabs"));
306  TabWidget::OpenUrlIn tab = TabWidget::OpenUrlIn(settings.value(QLatin1String("openLinksFromAppsIn"), TabWidget::NewSelectedTab).toInt());
307  settings.endGroup();
308  if (QUrl(message) == m_lastAskedUrl
309  && m_lastAskedUrlDateTime.addSecs(10) > QDateTime::currentDateTime()) {
310  qWarning() << "Possible recursive openUrl called, ignoring url:" << m_lastAskedUrl;
311  return;
312  }
313  mainWindow()->tabWidget()->loadString(message, tab);
314  return;
315  } else {
316  if (message.startsWith(QLatin1String("endorphin://getwinid"))) {
317 #ifdef Q_OS_WIN
318  QString winid = QString(QLatin1String("%1")).arg((qlonglong)mainWindow()->winId());
319 #else
320  mainWindow()->show();
321  mainWindow()->setFocus();
322  mainWindow()->raise();
323  mainWindow()->activateWindow();
324  alert(mainWindow());
325  QString winid;
326 #endif
327 #ifdef BROWSERAPPLICATION_DEBUG
328  qDebug() << "BrowserApplication::" << __FUNCTION__ << "sending win id" << winid << mainWindow()->winId();
329 #endif
330  QString message = QLatin1String("endorphin://winid/") + winid;
331  socket->write(message.toUtf8());
332  socket->waitForBytesWritten();
333  return;
334  }
335 
336  if (message.startsWith(QLatin1String("endorphin://winid"))) {
337  QString winid = message.mid(21);
338 #ifdef BROWSERAPPLICATION_DEBUG
339  qDebug() << "BrowserApplication::" << __FUNCTION__ << "got win id:" << winid;
340 #endif
341 #ifdef Q_OS_WIN
342  WId wid = (WId)winid.toLongLong();
343  SetForegroundWindow(wid);
344 #endif
345  return;
346  }
347  QSettings settings;
348  settings.beginGroup(QLatin1String("tabs"));
349  TabWidget::OpenUrlIn tab = TabWidget::OpenUrlIn(settings.value(QLatin1String("openLinksFromAppsIn"), TabWidget::NewSelectedTab).toInt());
350  settings.endGroup();
351  if (QUrl(message) == m_lastAskedUrl
352  && m_lastAskedUrlDateTime.addSecs(10) > QDateTime::currentDateTime()) {
353  qWarning() << "Possible recursive openUrl called, ignoring url:" << m_lastAskedUrl;
354  return;
355  }
356  mainWindow()->tabWidget()->loadString(message, tab);
357  return;
358  }
359 }

361 void BrowserApplication::quitBrowser()
362 {
363  if (s_downloadManager && !downloadManager()->allowQuit())
364  return;
365 
366  if (QSettings().value(QLatin1String("tabs/confirmClosingMultipleTabs"), true).toBool()) {
367  clean();
368  int tabCount = 0;
369  for (int i = 0; i < m_mainWindows.count(); ++i) {
370  tabCount += m_mainWindows.at(i)->tabWidget()->count();
371  }
372 
373  if (tabCount > 1) {
374  QWidget *widget = mainWindow();
375  QApplication::alert(widget);
376  int ret = QMessageBox::warning(widget, QString(),
377  tr("There are %1 windows and %2 tabs open\n"
378  "Do you want to quit anyway?").arg(m_mainWindows.count()).arg(tabCount),
379  QMessageBox::Yes | QMessageBox::No,
380  QMessageBox::No);
381  if (ret == QMessageBox::No)
382  return;
383  }
384  }
385 
386  saveSession();
387  exit(0);
388 }

393 void BrowserApplication::postLaunch()
394 {
395  QString directory = QStandardPaths::writableLocation(QStandardPaths::CacheLocation);
396  if (directory.isEmpty())
397  directory = QDir::homePath() + QLatin1String("/.") + QCoreApplication::applicationName();
398  // QWebEngineSettings::setIconDatabasePath(directory);
399 
400  loadSettings();
401 
402  // newMainWindow() needs to be called in main() for this to happen
403  if (m_mainWindows.count() > 0) {
404  QSettings settings;
405  settings.beginGroup(QLatin1String("MainWindow"));
406  int startup = settings.value(QLatin1String("startupBehavior")).toInt();
407  QStringList args = QCoreApplication::arguments();
408 
409  if (args.count() > 1) {
410  QString argumentUrl = parseArgumentUrl(args.last());
411  switch (startup) {
412  case 2: {
413  restoreLastSession();
414  mainWindow()->tabWidget()->loadString(argumentUrl, TabWidget::NewSelectedTab);
415  break;
416  }
417  default:
418  mainWindow()->tabWidget()->loadString(argumentUrl);
419  break;
420  }
421  } else {
422  switch (startup) {
423  case 0:
424  mainWindow()->goHome();
425  break;
426  case 1:
427  break;
428  case 2:
429  restoreLastSession();
430  break;
431  }
432  }
433  }
434  BrowserApplication::historyManager();
435 }

437 void BrowserApplication::loadSettings()
438 {
439  QSettings settings;
440  settings.beginGroup(QLatin1String("websettings"));
441 
442  QWebEngineSettings *defaultSettings = QWebEngineSettings::globalSettings();
443  QWebEngineProfile *defaultProfile = QWebEngineProfile::defaultProfile();
444  QString standardFontFamily = defaultSettings->fontFamily(QWebEngineSettings::StandardFont);
445  int standardFontSize = defaultSettings->fontSize(QWebEngineSettings::DefaultFontSize);
446  QFont standardFont = QFont(standardFontFamily, standardFontSize);
447  standardFont = settings.value(QLatin1String("standardFont"), standardFont).value<QFont>();
448  defaultSettings->setFontFamily(QWebEngineSettings::StandardFont, standardFont.family());
449  defaultSettings->setFontSize(QWebEngineSettings::DefaultFontSize, standardFont.pointSize());
450  int minimumFontSize = settings.value(QLatin1String("minimumFontSize"),
451  defaultSettings->fontSize(QWebEngineSettings::MinimumFontSize)).toInt();
452  defaultSettings->setFontSize(QWebEngineSettings::MinimumFontSize, minimumFontSize);
453 
454  QString fixedFontFamily = defaultSettings->fontFamily(QWebEngineSettings::FixedFont);
455  int fixedFontSize = defaultSettings->fontSize(QWebEngineSettings::DefaultFixedFontSize);
456  QFont fixedFont = QFont(fixedFontFamily, fixedFontSize);
457  fixedFont = settings.value(QLatin1String("fixedFont"), fixedFont).value<QFont>();
458  defaultSettings->setFontFamily(QWebEngineSettings::FixedFont, fixedFont.family());
459  defaultSettings->setFontSize(QWebEngineSettings::DefaultFixedFontSize, fixedFont.pointSize());
460 
461  defaultSettings->setAttribute(QWebEngineSettings::JavascriptCanOpenWindows, !(settings.value(QLatin1String("blockPopupWindows"), true).toBool()));
462  defaultSettings->setAttribute(QWebEngineSettings::JavascriptEnabled, settings.value(QLatin1String("enableJavascript"), true).toBool());
463  defaultSettings->setAttribute(QWebEngineSettings::AutoLoadImages, settings.value(QLatin1String("enableImages"), true).toBool());
464  defaultSettings->setAttribute(QWebEngineSettings::LocalStorageEnabled, settings.value(QLatin1String("enableLocalStorage"), true).toBool());
465  //defaultSettings->setAttribute(QWebEngineSettings::DeveloperExtrasEnabled, settings.value(QLatin1String("enableInspector"), false).toBool());
466  defaultSettings->setAttribute(QWebEngineSettings::DnsPrefetchEnabled, true);
467 
468  QString css = settings.value(QLatin1String("userStyleSheet")).toString();
469  setUserStyleSheet(defaultProfile, css, mainWindow());
470 
471 // int maximumPagesInCache = settings.value(QLatin1String("maximumPagesInCache"), 3).toInt();
472 // QWebEngineSettings::globalSettings()->setMaximumPagesInCache(maximumPagesInCache);
473 
474  settings.endGroup();
475 }

508 void BrowserApplication::saveSession()
509 {
510  if (quitting)
511  return;
512  QSettings settings;
513  settings.beginGroup(QLatin1String("MainWindow"));
514  settings.setValue(QLatin1String("restoring"), false);
515  settings.endGroup();
516 
517  /*
518  QWebEngineSettings *globalSettings = QWebEngineSettings::globalSettings();
519  if (globalSettings->testAttribute(QWebEngineSettings::PrivateBrowsingEnabled))
520  return;
521  */
522  clean();
523 
524  settings.beginGroup(QLatin1String("sessions"));
525 
526  int version = 2;
527 
528  QByteArray data;
529  QBuffer buffer(&data);
530  QDataStream stream(&buffer);
531  buffer.open(QIODevice::WriteOnly);
532 
533  stream << qint32(BrowserApplicationMagic);
534  stream << qint32(version);
535 
536  stream << qint32(m_mainWindows.count());
537  for (int i = 0; i < m_mainWindows.count(); ++i)
538  stream << m_mainWindows.at(i)->saveState();
539  settings.setValue(QLatin1String("lastSession"), data);
540  settings.endGroup();
541 }

548 bool BrowserApplication::restoreLastSession()
549 {
550  {
551  QSettings settings;
552  settings.beginGroup(QLatin1String("MainWindow"));
553  if (settings.value(QLatin1String("restoring"), false).toBool()) {
554  QMessageBox::StandardButton result = QMessageBox::question(nullptr, tr("Restore failed"),
555  tr("Endorphin crashed while trying to restore this session. Should I try again?"), QMessageBox::Yes | QMessageBox::No);
556  if (result == QMessageBox::No)
557  return false;
558  }
559  // saveSession will be called by an AutoSaver timer from the set tabs
560  // and in saveSession we will reset this flag back to false
561  settings.setValue(QLatin1String("restoring"), true);
562  }
563  int version = 2;
564  QList<QByteArray> windows;
565  QBuffer buffer(&m_lastSession);
566  QDataStream stream(&buffer);
567  buffer.open(QIODevice::ReadOnly);
568 
569  qint32 marker;
570  qint32 v;
571  stream >> marker;
572  stream >> v;
573  if (marker != BrowserApplicationMagic || v != version)
574  return false;
575 
576  qint32 windowCount;
577  stream >> windowCount;
578  for (qint32 i = 0; i < windowCount; ++i) {
579  QByteArray windowState;
580  stream >> windowState;
581  windows.append(windowState);
582  }
583  for (int i = 0; i < windows.count(); ++i) {
584  BrowserMainWindow *newWindow = nullptr;
585  if (i == 0 && m_mainWindows.count() >= 1) {
586  newWindow = mainWindow();
587  } else {
588  newWindow = newMainWindow();
589  }
590  newWindow->restoreState(windows.at(i));
591  }
592  return true;
593 }
