298 QByteArray BrowserMainWindow::saveState(bool withTabs) const
299 {
300  int version = 3;
301  QByteArray data;
302  QDataStream stream(&data, QIODevice::WriteOnly);
303 
304  stream << qint32(BrowserMainWindowMagic);
305  stream << qint32(version);
306 
307  // save the normal size so exiting fullscreen/maximize will work reasonably
308  stream << normalGeometry().size();
309  stream << !m_navigationBar->isHidden(); // DEAD
310  stream << !m_bookmarksToolbar->isHidden(); // DEAD
311  stream << !statusBar()->isHidden();
312  if (withTabs)
313  stream << tabWidget()->saveState();
314  else
315  stream << QByteArray();
316  stream << m_navigationSplitter->saveState();
317  stream << m_tabWidget->tabBar()->showTabBarWhenOneTab();
318 
319  stream << qint32(toolBarArea(m_navigationBar));
320  stream << qint32(toolBarArea(m_bookmarksToolbar));
321 
322  // version 3
323  stream << isMaximized();
324  stream << isFullScreen();
325  stream << menuBar()->isVisible();
326  stream << m_menuBarVisible; // DEAD
327  stream << m_statusBarVisible;
328 
329  stream << QMainWindow::saveState();
330 
331  return data;
332 }

334 bool BrowserMainWindow::restoreState(const QByteArray &state)
335 {
336  QByteArray sd = state;
337  QDataStream stream(&sd, QIODevice::ReadOnly);
338  if (stream.atEnd())
339  return false;
340 
341  qint32 marker;
342  qint32 version;
343  stream >> marker;
344  stream >> version;
345  if (marker != BrowserMainWindowMagic || !(version == 2 || version == 3))
346  return false;
347 
348  QSize size;
349  bool showToolbarDEAD;
350  bool showBookmarksBarDEAD;
351  bool showStatusbar;
352  QByteArray tabState;
353  QByteArray splitterState;
354  bool showTabBarWhenOneTab;
355  qint32 navigationBarLocation;
356  qint32 bookmarkBarLocation;
357  bool maximized;
358  bool fullScreen;
359  bool showMenuBar;
360  QByteArray qMainWindowState;
361 
362  stream >> size;
363  stream >> showToolbarDEAD;
364  stream >> showBookmarksBarDEAD;
365  stream >> showStatusbar;
366  stream >> tabState;
367  stream >> splitterState;
368  stream >> showTabBarWhenOneTab;
369  stream >> navigationBarLocation;
370  stream >> bookmarkBarLocation;
371 
372  if (version >= 3) {
373  stream >> maximized;
374  stream >> fullScreen;
375  stream >> showMenuBar;
376  stream >> showMenuBar; // m_menuBarVisible DEAD
377  stream >> m_statusBarVisible;
378  stream >> qMainWindowState;
379  } else {
380  maximized = false;
381  fullScreen = false;
382  showMenuBar = true;
383  m_statusBarVisible = showStatusbar;
384  }
385 
386  if (size.isValid())
387  resize(size);
388 
389  if (maximized)
390  setWindowState(windowState() | Qt::WindowMaximized);
391  if (fullScreen) {
392  setWindowState(windowState() | Qt::WindowFullScreen);
393  m_viewFullScreenAction->setChecked(true);
394  }
395 
396  menuBar()->setVisible(showMenuBar);
397  m_menuBarVisible = showMenuBar;
398 
399  statusBar()->setVisible(showStatusbar);
400 
401  m_navigationSplitter->restoreState(splitterState);
402 
403  tabWidget()->restoreState(tabState);
404 
405  m_tabWidget->tabBar()->setShowTabBarWhenOneTab(showTabBarWhenOneTab);
406 
407  if (qMainWindowState.isEmpty()) {
408  m_navigationBar->setVisible(showToolbarDEAD);
409  m_bookmarksToolbar->setVisible(showBookmarksBarDEAD);
410  Qt::ToolBarArea navigationArea = Qt::ToolBarArea(navigationBarLocation);
411  if (navigationArea != Qt::TopToolBarArea && navigationArea != Qt::NoToolBarArea)
412  addToolBar(navigationArea, m_navigationBar);
413  Qt::ToolBarArea bookmarkArea = Qt::ToolBarArea(bookmarkBarLocation);
414  if (bookmarkArea != Qt::TopToolBarArea && bookmarkArea != Qt::NoToolBarArea)
415  addToolBar(bookmarkArea, m_bookmarksToolbar);
416  } else {
417  QMainWindow::restoreState(qMainWindowState);
418  }
419 
420 #if defined(Q_WS_MAC)
421  m_bookmarksToolbarFrame->setVisible(m_bookmarksToolbar->isVisible());
422 #endif
423 
424  return true;
425 }

869  foreach (const int &mib, mibs) {
870  QString codec = QLatin1String(QTextCodec::codecForMib(mib)->name());
871  codecs.append(codec);
872  }
873  codecs.sort();
874 
875  QString defaultTextEncoding = QWebEngineSettings::globalSettings()->defaultTextEncoding();
876  currentCodec = codecs.indexOf(defaultTextEncoding);

878  QAction *defaultEncoding = m_viewTextEncodingMenu->addAction(tr("Default"));

881  if (currentCodec == -1)
882  defaultEncoding->setChecked(true);
883  m_viewTextEncodingMenu->addSeparator();
884 
885  for (int i = 0; i < codecs.count(); ++i) {
886  const QString &codec = codecs.at(i);
887  QAction *action = m_viewTextEncodingMenu->addAction(codec);
888  action->setData(codec);
889  action->setCheckable(true);
890  if (currentCodec == i)
891  action->setChecked(true);
