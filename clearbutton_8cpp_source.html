58 void ClearButton::paintEvent(QPaintEvent *event)
59 {
60  Q_UNUSED(event);
61  QPainter painter(this);
62 
63  if (!m_styleImage.isNull()) {
64  int x = (width() - m_styleImage.width()) / 2 - 1;
65  int y = (height() - m_styleImage.height()) / 2 - 1;
66  painter.drawImage(x, y, m_styleImage);
67  return;
68  }
69 
70  // Fall back to boring circle X
71  painter.setRenderHint(QPainter::Antialiasing, true);
72 
73  QPalette p = palette();
74  QColor circleColor = isDown() ? p.color(QPalette::Dark) : p.color(QPalette::Mid);
75  QColor xColor = p.color(QPalette::Window);
76 
77  // draw circle
78  painter.setBrush(circleColor);
79  painter.setPen(circleColor);
80  int padding = width() / 5;
81  int circleRadius = width() - (padding * 2);
82  painter.drawEllipse(padding, padding, circleRadius, circleRadius);
83 
84  // draw X
85  painter.setPen(xColor);
86  padding = padding * 2;
87  painter.drawLine(padding, padding, width() - padding, width() - padding);
88  painter.drawLine(padding, height() - padding, width() - padding, padding);
89 }
