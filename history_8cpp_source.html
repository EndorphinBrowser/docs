143 QVariant HistoryModel::data(const QModelIndex &index, int role) const
144 {
145  QList<HistoryEntry> lst = m_history->history();
146  if (index.row() < 0 || index.row() >= lst.size())
147  return QVariant();
148 
149  const HistoryEntry &item = lst.at(index.row());
150  switch (role) {
151  case DateTimeRole:
152  return item.dateTime;
153  case DateRole:
154  return item.dateTime.date();
155  case UrlRole:
156  return QUrl(item.url);
157  case UrlStringRole:
158  return item.url;
159  case TitleRole:
160  return item.userTitle();
161  case Qt::DisplayRole:
162  case Qt::EditRole: {
163  switch (index.column()) {
164  case 0:
165  return item.userTitle();
166  case 1:
167  return item.url;
168  }
169  }
170 #ifndef NO_BROWSERAPPLICATION
171  case Qt::DecorationRole:
172  if (index.column() == 0) {
173  return BrowserApplication::instance()->icon(item.url);
174  }
175 #endif
176  }
177  return QVariant();
178 }

190 bool HistoryModel::removeRows(int row, int count, const QModelIndex &parent)
191 {
192  if (parent.isValid())
193  return false;
194  int lastRow = row + count - 1;
195  beginRemoveRows(parent, row, lastRow);
196  QList<HistoryEntry> lst = m_history->history();
197  for (int i = lastRow; i >= row; --i)
198  lst.removeAt(i);
199  disconnect(m_history, SIGNAL(historyReset()), this, SLOT(historyReset()));
200  m_history->setHistory(lst);
201  connect(m_history, SIGNAL(historyReset()), this, SLOT(historyReset()));
202  endRemoveRows();

231 int HistoryMenuModel::rowCount(const QModelIndex &parent) const
232 {
233  if (parent.column() > 0)
234  return 0;
235 
236  if (!parent.isValid()) {
237  int folders = sourceModel()->rowCount();
238  int bumpedItems = bumpedRows();
239  if (bumpedItems <= MOVEDROWS
240  && bumpedItems == sourceModel()->rowCount(sourceModel()->index(0, 0)))
241  --folders;
242  return bumpedItems + folders;
243  }
244 
245  if (parent.internalId() == -1) {
246  if (parent.row() < bumpedRows())
247  return 0;
248  }
249 
250  QModelIndex idx = mapToSource(parent);
251  int defaultCount = sourceModel()->rowCount(idx);
252  if (idx == sourceModel()->index(0, 0))
253  return defaultCount - bumpedRows();
254  return defaultCount;
255 }

270 QModelIndex HistoryMenuModel::mapToSource(const QModelIndex &proxyIndex) const
271 {
272  if (!proxyIndex.isValid())
273  return QModelIndex();
274 
275  if (proxyIndex.internalId() == -1) {
276  int bumpedItems = bumpedRows();
277  if (proxyIndex.row() < bumpedItems)
278  return m_treeModel->index(proxyIndex.row(), proxyIndex.column(), m_treeModel->index(0, 0));
279  if (bumpedItems <= MOVEDROWS && bumpedItems == sourceModel()->rowCount(m_treeModel->index(0, 0)))
280  --bumpedItems;
281  return m_treeModel->index(proxyIndex.row() - bumpedItems, proxyIndex.column());
282  }
283 
284  QModelIndex historyIndex = m_treeModel->sourceModel()->index(proxyIndex.internalId(), proxyIndex.column());
285  QModelIndex treeIndex = m_treeModel->mapFromSource(historyIndex);
286  return treeIndex;
287 }

289 QModelIndex HistoryMenuModel::index(int row, int column, const QModelIndex &parent) const
290 {
291  if (row < 0
292  || column < 0 || column >= columnCount(parent)
293  || parent.column() > 0)
294  return QModelIndex();
295  if (!parent.isValid())
296  return createIndex(row, column, -1);
297 
298  QModelIndex treeIndexParent = mapToSource(parent);
299 
300  int bumpedItems = 0;
301  if (treeIndexParent == m_treeModel->index(0, 0))
302  bumpedItems = bumpedRows();
303  QModelIndex treeIndex = m_treeModel->index(row + bumpedItems, column, treeIndexParent);
304  QModelIndex historyIndex = m_treeModel->mapToSource(treeIndex);
305  int historyRow = historyIndex.row();
306  if (historyRow == -1)
307  historyRow = treeIndex.row();
308  return createIndex(row, column, historyRow);
309 }

312 {

315  return QModelIndex();
316

318  QModelIndex treeIndex = m_treeModel->mapFromSource(historyIndex);
319  QModelIndex treeIndexParent = treeIndex.parent();
320 
321  int sr = m_treeModel->mapToSource(treeIndexParent).row();

323  if (bumpedItems <= MOVEDROWS && bumpedItems == sourceModel()->rowCount(sourceModel()->index(0, 0)))

629 void HistoryFilterModel::load() const
630 {
631  if (m_loaded)
632  return;
633  m_filteredRows.clear();
634  m_historyHash.clear();
635  m_historyHash.reserve(sourceModel()->rowCount());
636  m_scaleTime = QDateTime::currentDateTime();
637  for (int i = 0; i < sourceModel()->rowCount(); ++i) {
638  QModelIndex idx = sourceModel()->index(i, 0);
639  QString url = idx.data(HistoryModel::UrlStringRole).toString();
640  if (!m_historyHash.contains(url)) {
641  int sourceOffset = sourceModel()->rowCount() - i;
642  m_filteredRows.append(HistoryData(sourceOffset, frecencyScore(idx)));
643  m_historyHash.insert(url, sourceOffset);
644  } else {
645  // we already know about this url: just increment its frecency score
646  QList<HistoryData>::iterator pos = std::lower_bound(m_filteredRows.begin(),
647  m_filteredRows.end(), HistoryData(m_historyHash[url], -1));
648  Q_ASSERT(pos != m_filteredRows.end());
649  pos->frecency += frecencyScore(idx);
650  }
651  }
652  m_loaded = true;
653 }

655 void HistoryFilterModel::sourceRowsInserted(const QModelIndex &parent, int start, int end)
656 {
657  Q_ASSERT(start == end && start == 0);
658  Q_UNUSED(end);
659  if (!m_loaded)
660  return;
661  QModelIndex idx = sourceModel()->index(start, 0, parent);
662  QString url = idx.data(HistoryModel::UrlStringRole).toString();
663  int currentFrecency = 0;
664  if (m_historyHash.contains(url)) {
665  QList<HistoryData>::iterator pos = std::lower_bound(m_filteredRows.begin(),
666  m_filteredRows.end(), HistoryData(m_historyHash[url], -1));
