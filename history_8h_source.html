155  struct HistoryData {
156  int tailOffset;
157  int frecency;
158 
159  HistoryData(int off, int f = 0) : tailOffset(off), frecency(f) { }
160 
161  bool operator==(const HistoryData &other) const {
162  return (tailOffset == other.tailOffset)
163  && (frecency == -1 || other.frecency == -1 || frecency == other.frecency);
164  }
165  bool operator!=(const HistoryData &other) const {
166  return !(*this == other);
167  }
168  // like the actual history entries, our index mapping data is sorted in reverse
169  bool operator<(const HistoryData &other) const {
170  return (tailOffset > other.tailOffset);
171  }
172  };
173  int frecencyScore(const QModelIndex &sourceIndex) const;
174 
175  mutable QList<HistoryData> m_filteredRows;
176  mutable QHash<QString, int> m_historyHash;
177  mutable bool m_loaded;
178  mutable QDateTime m_scaleTime;
179 };
