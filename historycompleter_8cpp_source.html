69 QVariant HistoryCompletionModel::data(const QModelIndex &index, int role) const
70 {
71  // if we are valid, tell QCompleter that everything we have filtered matches
72  // what the user typed; if not, nothing matches
73  if (role == HistoryCompletionRole && index.isValid()) {
74  if (isValid())
75  return QLatin1String("a");
76  else
77  return QLatin1String("b");
78  }
79 
80  if (role == Qt::FontRole && index.column() == 1) {
81  QFont font = qvariant_cast<QFont>(QSortFilterProxyModel::data(index, role));
82  font.setWeight(QFont::Light);
83  return font;
84  }
85 
86  if (role == Qt::DisplayRole)
87  role = (index.column() == 0) ? HistoryModel::UrlStringRole : HistoryModel::TitleRole;
88 
89  return QSortFilterProxyModel::data(index, role);

124 bool HistoryCompletionModel::filterAcceptsRow(int source_row, const QModelIndex &source_parent) const
125 {
126  // do a case-insensitive substring match against both the url and title;
127  // we have also made sure that the user doesn't accidentally use regexp
128  // metacharacters
129  QModelIndex idx = sourceModel()->index(source_row, 0, source_parent);
130  QString url = sourceModel()->data(idx, HistoryModel::UrlStringRole).toString();
131 
132  if (m_searchMatcher.indexIn(url) != -1)
133  return true;
134 
135  QString title = sourceModel()->data(idx, HistoryModel::TitleRole).toString();
136 
137  if (m_searchMatcher.indexIn(title) != -1)
138  return true;
139 
140  return false;
141 }

143 bool HistoryCompletionModel::lessThan(const QModelIndex &left, const QModelIndex &right) const
144 {
145  // We give a bonus to hits that match on a word boundary so that e.g. "dot.kde.org"
146  // is a better result for typing "dot" than "slashdot.org". However, we only look
147  // for the string in the host name, not the entire url, since while it makes sense
148  // to e.g. give "www.phoronix.com" a bonus for "ph", it does _not_ make sense to
149  // give "www.yadda.com/foo.php" the bonus.
150  int frecency_l = sourceModel()->data(left, HistoryFilterModel::FrecencyRole).toInt();
151  QString url_l = sourceModel()->data(left, HistoryModel::UrlRole).toUrl().host();
152  QString title_l = sourceModel()->data(left, HistoryModel::TitleRole).toString();
153 
154  if (m_wordMatcher.indexIn(url_l) != -1 || m_wordMatcher.indexIn(title_l) != -1)
155  frecency_l *= 2;
156 
157  int frecency_r = sourceModel()->data(right, HistoryFilterModel::FrecencyRole).toInt();
158  QString url_r = sourceModel()->data(right, HistoryModel::UrlRole).toUrl().host();
159  QString title_r = sourceModel()->data(right, HistoryModel::TitleRole).toString();
160  if (m_wordMatcher.indexIn(url_r) != -1 || m_wordMatcher.indexIn(title_r) != -1)
161  frecency_r *= 2;
162 
163  // sort results in descending frecency-derived score
164  return (frecency_r < frecency_l);
165 }

202 QStringList HistoryCompleter::splitPath(const QString &path) const
203 {
204  if (path == m_searchString)
205  return QStringList() << QLatin1String("a");
206 
207  // queue an update to our search string
208  // We will wait a bit so that if the user is quickly typing,
209  // we don't try to complete until they pause.
210  if (m_filterTimer.isActive())
211  m_filterTimer.stop();
212  m_filterTimer.start(150);
213 
214  // if the previous search results are not a superset of
215  // the current search results, tell the model that it is not valid yet
216  if (!path.startsWith(m_searchString)) {
217  HistoryCompletionModel *completionModel = qobject_cast<HistoryCompletionModel*>(model());
218  Q_ASSERT(completionModel);
219  completionModel->setValid(false);
220  }
221 
222  m_searchString = path;
223 
224  // the actual filtering is done by the HistoryCompletionModel; we just
225  // return a short dummy here so that QCompleter thinks we match everything
226  return QStringList() << QLatin1String("a");
227 }

229 bool HistoryCompleter::eventFilter(QObject *obj, QEvent *event)
230 {
231  if (event->type() == QEvent::KeyPress && popup()->isVisible()) {
232  QKeyEvent *keyEvent = static_cast<QKeyEvent*>(event);
233  if (keyEvent->key() == Qt::Key_Tab) {
234  QKeyEvent *newEvent = new QKeyEvent(QEvent::KeyPress,
235  Qt::Key_Down,
236  keyEvent->modifiers(),
237  QString());
238 
239  if (!QCompleter::eventFilter(obj, newEvent))
240  obj->event(newEvent);
241  return true;
242  } else if (keyEvent->key() == Qt::Key_Backtab) {
243  QKeyEvent *newEvent = new QKeyEvent(QEvent::KeyPress,
244  Qt::Key_Up,
245  keyEvent->modifiers(),
246  keyEvent->text(),
247  keyEvent->isAutoRepeat(),
248  keyEvent->count());
249 
250  if (!QCompleter::eventFilter(obj, newEvent))
251  obj->event(newEvent);
252  return true;
253  } else if (keyEvent->key() == Qt::Key_Escape) {
254  popup()->hide();
255  }
256  }
257  return QCompleter::eventFilter(obj, event);
258 }
