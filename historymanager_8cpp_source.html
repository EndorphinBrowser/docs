188 void HistoryManager::checkForExpired()
189 {
190  if (m_daysToExpire < 0 || m_history.isEmpty())
191  return;
192 
193  QDateTime now = QDateTime::currentDateTime();
194  int nextTimeout = 0;
195 
196  while (!m_history.isEmpty()) {
197  QDateTime checkForExpired = m_history.last().dateTime;
198  checkForExpired.setDate(checkForExpired.date().addDays(m_daysToExpire));
199  if (now.daysTo(checkForExpired) > 7) {
200  // check at most in a week to prevent int overflows on the timer
201  nextTimeout = 7 * 86400;
202  } else {
203  nextTimeout = now.secsTo(checkForExpired);
204  }
205  if (nextTimeout > 0)
206  break;
207  HistoryEntry item = m_history.takeLast();
208  // remove from saved file also
209  m_lastSavedUrl.clear();
210  emit entryRemoved(item);
211  }
212 
213  if (nextTimeout > 0)
214  m_expiredTimer.start(nextTimeout * 1000);
215 }

297 void HistoryManager::load()
298 {
299  loadSettings();
300 
301  QFile historyFile(HistoryManager::dataFilePath(QLatin1String("history")));
302 
303  if (!historyFile.exists())
304  return;
305  if (!historyFile.open(QFile::ReadOnly)) {
306  qWarning() << "Unable to open history file" << historyFile.fileName();
307  return;
308  }
309 
310  QList<HistoryEntry> list;
311  QDataStream in(&historyFile);
312  // Double check that the history file is sorted as it is read in
313  bool needToSort = false;
314  HistoryEntry lastInsertedItem;
315  QByteArray data;
316  QDataStream stream;
317  QBuffer buffer;
318  QString string;
319  stream.setDevice(&buffer);
320  while (!historyFile.atEnd()) {
321  in >> data;
322  buffer.close();
323  buffer.setBuffer(&data);
324  buffer.open(QIODevice::ReadOnly);
325  quint32 ver;
326  stream >> ver;
327  if (ver != HISTORY_VERSION)
328  continue;
329  HistoryEntry item;
330  stream >> string;
331  item.url = atomicString(string);
332  stream >> item.dateTime;
333  stream >> string;
334  item.title = atomicString(string);
335 
336  if (!item.dateTime.isValid())
337  continue;
338 
339  if (item == lastInsertedItem) {
340  if (lastInsertedItem.title.isEmpty() && !list.isEmpty())
341  list[0].title = item.title;
342  continue;
343  }
344 
345  if (!needToSort && !list.isEmpty() && lastInsertedItem < item)
346  needToSort = true;
347 
348  list.prepend(item);
349  lastInsertedItem = item;
350  }
351  if (needToSort)
352  std::sort(list.begin(), list.end());
353 
354  setHistory(list, true);
355 
356  // If we had to sort re-write the whole history sorted
357  if (needToSort) {
358  m_lastSavedUrl.clear();
359  m_saveTimer->changeOccurred();
360  }
361 }

372 void HistoryManager::save()
373 {
374  QSettings settings;
375  settings.beginGroup(QLatin1String("history"));
376  settings.setValue(QLatin1String("historyLimit"), m_daysToExpire);
377 
378  bool saveAll = m_lastSavedUrl.isEmpty();
379  int first = m_history.count() - 1;
380  if (!saveAll) {
381  // find the first one to save
382  for (int i = 0; i < m_history.count(); ++i) {
383  if (m_history.at(i).url == m_lastSavedUrl) {
384  first = i - 1;
385  break;
386  }
387  }
388  }
389  if (first == m_history.count() - 1)
390  saveAll = true;
391 
392  QFile historyFile(HistoryManager::dataFilePath(QLatin1String("history")));
393 
394  // When saving everything use a temporary file to prevent possible data loss.
395  QTemporaryFile tempFile;
396  tempFile.setAutoRemove(false);
397  bool open = false;
398  if (saveAll) {
399  open = tempFile.open();
400  } else {
401  open = historyFile.open(QFile::Append);
402  }
403 
404  if (!open) {
405  qWarning() << "Unable to open history file for saving"
406  << (saveAll ? tempFile.fileName() : historyFile.fileName());
407  return;
408  }
409 
410  QDataStream out(saveAll ? &tempFile : &historyFile);
411  for (int i = first; i >= 0; --i) {
412  QByteArray data;
413  QDataStream stream(&data, QIODevice::WriteOnly);
414  HistoryEntry item = m_history.at(i);
415  stream << HISTORY_VERSION << item.url << item.dateTime << item.title;
416  out << data;
417  }
418  tempFile.close();
419 
420  if (saveAll) {
421  if (historyFile.exists() && !historyFile.remove())
422  qWarning() << "History: error removing old history." << historyFile.errorString();
423  if (!tempFile.copy(historyFile.fileName()))
424  qWarning() << "History: error moving new history over old." << tempFile.errorString() << historyFile.fileName();
425  if (!tempFile.remove())
426  qWarning() << "History: error deleting temporary file." << tempFile.errorString() << tempFile.fileName();
427  }
428  m_lastSavedUrl = m_history.value(0).url;
429 }
