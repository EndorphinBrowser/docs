71 class HistoryEntry
72 {
73 public:
74  HistoryEntry() {}
75  HistoryEntry(const QString &u,
76  const QDateTime &d = QDateTime(), const QString &t = QString())
77  : url(u), title(t), dateTime(d) {}
78 
79  inline bool operator==(const HistoryEntry &other) const {
80  return other.title == title
81  && other.url == url && other.dateTime == dateTime;
82  }
83 
84  // history is sorted in reverse
85  inline bool operator <(const HistoryEntry &other) const
86  { return dateTime > other.dateTime; }
