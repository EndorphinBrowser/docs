82 bool LanguageManager::isLanguageAvailable(const QString &language) const
83 {
84 #ifdef LANGUAGEMANAGER_DEBUG
85  qDebug() << "LanguageManager::" << __FUNCTION__ << language;
86 #endif
87  if (language.isEmpty())
88  return true;
89 
90  // optimization so we don't have to load all the languages
91  if (!m_loaded) {
92  foreach (const QString &dir, m_localeDirectories) {
93  QString file = dir + QLatin1Char('/') + language + QLatin1String(".qm");
94  if (QFile::exists(file))
95  return true;
96  }
97  }
98 
99  return !(convertStringToLanguageFile(language).isEmpty());
100 }

102 // Return an empty string if we do not have the language file for string
103 // If we don't have an exact ma
104 // - Fall back to country
105 // - Fall back to the first country_ match if there is one
106 QString LanguageManager::convertStringToLanguageFile(const QString &string) const
107 {
108 #ifdef LANGUAGEMANAGER_DEBUG
109  qDebug() << "LanguageManager::" << __FUNCTION__ << string;
110 #endif
111  loadAvailableLanguages();
112  if (m_languages.contains(string))
113  return string;
114  QLocale locale(string);
115  QString fallback = locale.name().split(QLatin1Char('_')).value(0);
116  if (!string.contains(fallback))
117  return QString();
118 #ifdef LANGUAGEMANAGER_DEBUG
119  qDebug() << "LanguageManager::" << __FUNCTION__ << "fallback" << fallback;
120 #endif
121  if (m_languages.contains(fallback)) // fallback to the country
122  return fallback;
123 
124  // See if any language file matches the country
125  foreach (const QString &language, m_languages) {
126  QString country = QLocale(language).name().split(QLatin1Char('_')).value(0);
127  if (country == fallback)
128  return language;
129  }
130 
131  return QString();
132 }

134 bool LanguageManager::setCurrentLanguage(const QString &language)
135 {
136 #ifdef LANGUAGEMANAGER_DEBUG
137  qDebug() << "LanguageManager::" << __FUNCTION__ << language;
138 #endif
139  if (m_currentLanguage == language
140  || !isLanguageAvailable(language))
141  return false;
142 
143  m_currentLanguage = language;
144 
145  QSettings settings;
146  settings.beginGroup(QLatin1String("LanguageManager"));
147  settings.setValue(QLatin1String("language"), m_currentLanguage);
148 
149  if (m_currentLanguage.isEmpty()) {
150  delete m_appTranslator;
151  delete m_sysTranslator;
152  m_appTranslator = nullptr;
153  m_sysTranslator = nullptr;
154  emit languageChanged(currentLanguage());
155  return true;
156  }
157 
158  QTranslator *newAppTranslator = new QTranslator(this);
159  QString resourceDir = QLibraryInfo::location(QLibraryInfo::TranslationsPath);
160  QString languageFile = convertStringToLanguageFile(m_currentLanguage);
161  bool loaded = false;
162  foreach (const QString &dir, m_localeDirectories) {
163  loaded = newAppTranslator->load(languageFile, dir);
164  if (loaded)
165  break;
166  }
167 
168  QTranslator *newSysTranslator = new QTranslator(this);
169  QString translatorFileName = QLatin1String("qt_") + languageFile;
170  if (!newSysTranslator->load(translatorFileName, resourceDir)) {
171  delete newSysTranslator;
172  newSysTranslator = nullptr;
173  }
174 
175  if (!loaded) {
176  qWarning() << "Failed to load translation:" << currentLanguage();
177  delete newAppTranslator;
178  delete newSysTranslator;
179  return false;
180  }
181 
182  QLocale::setDefault(QLocale(m_currentLanguage));
183 
184  // A new language event is sent to all widgets in the application
185  // They need to catch it and re-translate
186  delete m_appTranslator;
187  delete m_sysTranslator;
188  qApp->installTranslator(newAppTranslator);
189  qApp->installTranslator(newSysTranslator);
190  m_appTranslator = newAppTranslator;
191  m_sysTranslator = newSysTranslator;
192  emit languageChanged(currentLanguage());
193  return true;
194 }

205 void LanguageManager::loadLanguageFromSettings()
206 {
207 #ifdef LANGUAGEMANAGER_DEBUG
208  qDebug() << "LanguageManager::" << __FUNCTION__;
209 #endif
210 
211  QSettings settings;
212  settings.beginGroup(QLatin1String("LanguageManager"));
213  if (settings.contains(QLatin1String("language"))) {
214  QString selectedLanguage = settings.value(QLatin1String("language")).toString();
215 #ifdef LANGUAGEMANAGER_DEBUG
216  qDebug() << "LanguageManager::" << __FUNCTION__ << "Loading language from settings" << selectedLanguage;
217 #endif
218  // When a translation fails to load remove it from the settings
219  // to prevent it from being loaded every time.
220  if (!setCurrentLanguage(selectedLanguage)) {
221 #ifdef LANGUAGEMANAGER_DEBUG
222  qDebug() << "LanguageManager::" << __FUNCTION__ << "Failed to load language";
223 #endif
224  settings.remove(QLatin1String("language"));
225  }
226  } else if (!currentLanguage().isEmpty()) {
227  setCurrentLanguage(currentLanguage());
228  }
229 }

231 void LanguageManager::chooseNewLanguage()
232 {
233 #ifdef LANGUAGEMANAGER_DEBUG
234  qDebug() << "LanguageManager::" << __FUNCTION__;
235 #endif
236  loadAvailableLanguages();
237  if (m_languages.isEmpty()) {
238  QMessageBox messageBox;
239  QLatin1String separator = QLatin1String(", ");
240  messageBox.setText(tr("No translation files are installed at %1.")
241  .arg(m_localeDirectories.join(separator)));
242  messageBox.setStandardButtons(QMessageBox::Ok);
243  messageBox.exec();
244  return;
245  }
246 
247  QStringList items;
248  int defaultItem = -1;
249  QString current = currentLanguage();
250  foreach (const QString &name, m_languages) {
251  QLocale locale(name);
252  QString string = QString(QLatin1String("%1, %2 (%3) %4"))
253  .arg(QLocale::languageToString(locale.language()))
254  .arg(QLocale::countryToString(locale.country()))
255  .arg(name)
256  // this is for pretty RTL support
257  .arg(QChar(0x200E)); // LRM = 0x200E
258  if (name == current)
259  defaultItem = items.count();
260  items << string;
261  }
262  items << QLatin1String("English (en_US)");
263  if (defaultItem == -1)
264  defaultItem = items.count() - 1;
265 
266  bool ok;
267  QString item = QInputDialog::getItem(nullptr,
268  tr("Choose language"),
269  tr("<p>You can run with a different language than<br>"
270  "the operating system default.</p>"
271  "<p>Please choose the language which should be used</p>"),
272  items, defaultItem, false, &ok);
273  if (!ok)
274  return;
275 
276  int selection = items.indexOf(item);
277  setCurrentLanguage(m_languages.value(selection));
278 }

283 void LanguageManager::loadAvailableLanguages() const
284 {
285  if (m_loaded)
286  return;
287  m_loaded = true;
288 #ifdef LANGUAGEMANAGER_DEBUG
289  qDebug() << "LanguageManager::" << __FUNCTION__;
290 #endif
291 
292  foreach (const QString &dir, m_localeDirectories) {
293  QDirIterator it(dir);
294  while (it.hasNext()) {
295  QString fileName = it.next();
296  if (!fileName.endsWith(QLatin1String(".qm")))
297  continue;
298  const QFileInfo info = it.fileInfo();
299  QString language = info.completeBaseName();
300  m_languages.append(language);
301  }
302  }
