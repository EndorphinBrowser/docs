151 int LineEdit::textMargin(WidgetPosition position) const
152 {
153  int spacing = m_rightLayout->spacing();
154  int w = 0;
155  if (position == LeftSide)
156  w = m_leftWidget->sizeHint().width();
157  else
158  w = m_rightWidget->sizeHint().width();

161  return w + spacing * 2;

174 void LineEdit::updateSideWidgetLocations()
175 {
176  QStyleOptionFrame opt;
177  initStyleOption(&opt);
178  QRect textRect = style()->subElementRect(QStyle::SE_LineEditContents, &opt, this);
179  int spacing = m_rightLayout->spacing();
180  textRect.adjust(spacing, 0, -spacing, 0);
181 
182  int left = textMargin(LineEdit::LeftSide);
183 
184  int midHeight = textRect.center().y() + 1;
185 
186  if (m_leftLayout->count() > 0) {
187  int leftHeight = midHeight - m_leftWidget->height() / 2;
188  int leftWidth = m_leftWidget->width();
189  if (leftWidth == 0)
190  leftHeight = midHeight - m_leftWidget->sizeHint().height() / 2;
191  m_leftWidget->move(textRect.x(), leftHeight);
192  }
193  textRect.setX(left);
194  textRect.setY(midHeight - m_rightWidget->sizeHint().height() / 2);
195  textRect.setHeight(m_rightWidget->sizeHint().height());
196  m_rightWidget->setGeometry(textRect);
197 }
