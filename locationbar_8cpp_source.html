83 void LocationBar::webViewUrlChanged(const QUrl &url)
84 {
85  if (hasFocus())
86  return;
87  if(!url.toString().startsWith(QString("qrc:/"))) {
88  setText(QString::fromUtf8(url.toEncoded()));
89  } else {
90  QUrl newurl = QUrl(url.toString());
91  QString url_tmp = newurl.toString().mid(5);
92  newurl = QUrl(QLatin1String("endorphin://") + url_tmp);
93  QString urlstr = newurl.toString();
94  if(urlstr.endsWith(QLatin1String(".html"))) {
95  urlstr.chop(5);
96  newurl = QUrl(urlstr);
97  }
98  setText(QString::fromUtf8(newurl.toEncoded()));
99  }
100  setCursorPosition(0);
101 }

103 void LocationBar::paintEvent(QPaintEvent *event)
104 {
105  QPalette p = palette();
106  QColor defaultBaseColor = QApplication::palette().color(QPalette::Base);
107  QColor backgroundColor = defaultBaseColor;
108  if (m_webView && m_webView->url().scheme() == QLatin1String("https")
109  && p.color(QPalette::Text).value() < 128) {
110  QColor lightYellow(248, 248, 210);
111  backgroundColor = lightYellow;
112  }
113 
114  // set the progress bar
115  if (m_webView) {
116  int progress = m_webView->progress();
117  if (progress == 0) {
118  p.setBrush(QPalette::Base, backgroundColor);
119  } else {
120  QColor loadingColor = QColor(116, 192, 250);
121  if (p.color(QPalette::Text).value() >= 128)
122  loadingColor = defaultBaseColor.darker(200);
123 
124  QLinearGradient gradient(0, 0, width(), 0);
125  gradient.setColorAt(0, loadingColor);
126  gradient.setColorAt(((double)progress)/100, backgroundColor);
127  p.setBrush(QPalette::Base, gradient);
128  }
129  setPalette(p);
130  }

132  LineEdit::paintEvent(event);

150 void LocationBar::keyPressEvent(QKeyEvent *event)
151 {
152  if (event->key() == Qt::Key_Escape && m_webView) {
153  setText(QString::fromUtf8(m_webView->url().toEncoded()));
154  selectAll();
155  return;
156  }
157 
158  QString currentText = text().trimmed();
159  if ((event->key() == Qt::Key_Enter || event->key() == Qt::Key_Return)
160  && !currentText.startsWith(QLatin1String("http://"), Qt::CaseInsensitive)) {
161  QString append;
162  if (event->modifiers() == Qt::ControlModifier)
163  append = QLatin1String(".com");
164  else if (event->modifiers() == (Qt::ControlModifier | Qt::ShiftModifier))
165  append = QLatin1String(".org");
166  else if (event->modifiers() == Qt::ShiftModifier)
167  append = QLatin1String(".net");
168  QUrl url(QLatin1String("http://") + currentText);
169  QString host = url.host();
170  if (!host.endsWith(append, Qt::CaseInsensitive)) {
171  host += append;
172  url.setHost(host);
173  setText(url.toString());
174  }
175  }
176 
177  LineEdit::keyPressEvent(event);
178 }

192

203
