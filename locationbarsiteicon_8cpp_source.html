64 void LocationBarSiteIcon::mouseMoveEvent(QMouseEvent *event)
65 {
66  if (event->buttons() == Qt::LeftButton
67  && (event->pos() - m_dragStartPos).manhattanLength() > QApplication::startDragDistance()
68  && m_webView) {
69  QDrag *drag = new QDrag(this);
70  QMimeData *mimeData = new QMimeData;
71  QString title = m_webView->title();
72  if (title.isEmpty())
73  title = QString::fromUtf8(m_webView->url().toEncoded());
74  mimeData->setText(title);
75  QList<QUrl> urls;
76  urls.append(m_webView->url());
77  mimeData->setUrls(urls);
78  const QPixmap *p = pixmap();
79  if (p)
80  drag->setPixmap(*p);
81  drag->setMimeData(mimeData);
82  drag->exec();
83  }
