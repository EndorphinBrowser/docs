183 void ModelMenu::createMenu(const QModelIndex &parent, int max, QMenu *parentMenu, QMenu *menu)
184 {
185  if (!menu) {
186  QVariant v;
187  v.setValue(parent);
188 
189  QString title = parent.data().toString();
190  ModelMenu *modelMenu = createBaseMenu();
191  // triggered goes all the way up the menu structure
192  disconnect(modelMenu, SIGNAL(triggered(QAction*)),
193  modelMenu, SLOT(actionTriggered(QAction*)));
194  modelMenu->setTitle(title);
195  QIcon icon = qvariant_cast<QIcon>(parent.data(Qt::DecorationRole));
196  modelMenu->setIcon(icon);
197  parentMenu->addMenu(modelMenu)->setData(v);
198  modelMenu->setRootIndex(parent);
199  modelMenu->setModel(m_model);
200  return;
201  }
202 
203  if (!m_model)
204  return;
205 
206  int end = m_model->rowCount(parent);
207  if (max != -1)
208  end = qMin(max, end);
209 
210 
211  for (int i = 0; i < end; ++i) {
212  QModelIndex idx = m_model->index(i, 0, parent);
213  if (m_model->hasChildren(idx)) {
214  createMenu(idx, -1, menu);
215  } else {
216  if (m_separatorRole != 0
217  && idx.data(m_separatorRole).toBool())
218  addSeparator();
219  else
220  menu->addAction(makeAction(idx));
221  }
222  if (menu == this && i == m_firstSeparator - 1)
223  addSeparator();
224  }
225 }

282 void ModelMenu::dropEvent(QDropEvent *event)
283 {
284  if (!m_model) {
285  QMenu::dropEvent(event);
286  return;
287  }
288 
289  int row;
290  QAction *action = actionAt(mapFromGlobal(QCursor::pos()));
291  QModelIndex index;
292  QModelIndex parentIndex = m_root;
293  if (!action) {
294  row = m_model->rowCount(m_root);
295  } else {
296  index = this->index(action);
297  Q_ASSERT(index.isValid());
298  row = index.row();
299 
300  if (m_model->hasChildren(index))
301  parentIndex = index;
302  }
303 
304  event->acceptProposedAction();
305  m_model->dropMimeData(event->mimeData(), event->dropAction(), row, 0, parentIndex);
306  QMenu::dropEvent(event);
307 }

325 void ModelMenu::mouseMoveEvent(QMouseEvent *event)
326 {
327  int manhattanLength = (event->pos() - m_dragStartPos).manhattanLength();
328 
329  if (manhattanLength <= QApplication::startDragDistance()) {
330  QMenu::mouseMoveEvent(event);
331  return;
332  }
333 
334  if (!(event->buttons() & Qt::LeftButton)) {
335  QMenu::mouseMoveEvent(event);
336  return;
337  }
338 
339  QAction *action = actionAt(m_dragStartPos);
340  QModelIndex idx = index(action);
341 
342  if (!idx.isValid()) {
343  QMenu::mouseMoveEvent(event);
344  return;
345  }
346 
347  QDrag *drag = new QDrag(this);
348  drag->setMimeData(m_model->mimeData((QModelIndexList() << idx)));
349  QRect actionRect = actionGeometry(action);
350  drag->setPixmap(this->grab(actionRect));
351 
352  if (drag->exec() == Qt::MoveAction) {
353  m_model->removeRow(idx.row(), m_root);
354 
355  QWidget *target = static_cast<QWidget*>(drag->target());
356  if (!this->isAncestorOf(target))
357  close();
358  else
359  aboutToShow();
360  }
361 }
