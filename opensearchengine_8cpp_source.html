34 /*!
35  \class OpenSearchEngine
36  \brief A class representing a single search engine described in OpenSearch format
37 
38  OpenSearchEngine is a class that represents a single search engine based on
39  the OpenSearch format.
40  For more information about the format, see http://www.opensearch.org/.
41 
42  Instances of the class hold all the data associated with the corresponding search
43  engines, such as name(), description() and also URL templates that are used
44  to construct URLs, which can be used later to perform search queries. Search engine
45  can also have an image, even an external one, in this case it will be downloaded
46  automatically from the network.
47 
48  OpenSearchEngine instances can be constructed from scratch but also read from
49  external sources and written back to them. OpenSearchReader and OpenSearchWriter
50  are the classes provided for reading and writing OpenSearch descriptions.
51 
52  Default constructed engines need to be filled with the necessary information before
53  they can be used to peform search requests. First of all, a search engine should have
54  the metadata including the name and the description.
55  However, the most important are URL templates, which are the construction of URLs
56  but can also contain template parameters, that are replaced with corresponding values
57  at the time of constructing URLs.
58 
59  There are two types of URL templates: search URL template and suggestions URL template.
60  Search URL template is needed for constructing search URLs, which point directly to
61  search results. Suggestions URL template is necessary to construct suggestion queries
62  URLs, which are then used for requesting contextual suggestions, a popular service
63  offered along with search results that provides search terms related to what has been
64  supplied by the user.
65 
66  Both types of URLs are constructed by the class, by searchUrl() and suggestionsUrl()
67  functions respectively. However, search requests are supposed to be performed outside
68  the class, while suggestion queries can be executed using the requestSuggestions()
69  method. The class will take care of peforming the network request and parsing the
70  JSON response.
71 
72  Both the image request and suggestion queries need network access. The class can
73  perform network requests on its own, though the client application needs to provide
74  a network access manager, which then will to be used for network operations.
75  Without that, both images delivered from remote locations and contextual suggestions
76  will be disabled.
77 
78  \sa OpenSearchReader, OpenSearchWriter
79 */

106 QString OpenSearchEngine::parseTemplate(const QString &searchTerm, const QString &searchTemplate)
107 {
108  QString language = QLocale().name();
109  // Simple conversion to RFC 3066.
110  language = language.replace(QLatin1Char('_'), QLatin1Char('-'));
111 
112  QString result = searchTemplate;
113  result.replace(QLatin1String("{count}"), QLatin1String("20"));
114  result.replace(QLatin1String("{startIndex}"), QLatin1String("0"));
115  result.replace(QLatin1String("{startPage}"), QLatin1String("0"));
116  result.replace(QLatin1String("{language}"), language);
117  result.replace(QLatin1String("{inputEncoding}"), QLatin1String("UTF-8"));
118  result.replace(QLatin1String("{outputEncoding}"), QLatin1String("UTF-8"));
119  result.replace(QRegExp(QLatin1String("\\{([^\\}]*:|)source\\??\\}")), QCoreApplication::applicationName());
120  result.replace(QLatin1String("{searchTerms}"), QLatin1String(QUrl::toPercentEncoding(searchTerm)));
121 
122  return result;
123 }

173 /*!
174  Constructs and returns a search URL with a given \a searchTerm.
175 
176  The URL template is processed according to the specification:
177  http://www.opensearch.org/Specifications/OpenSearch/1.1#OpenSearch_URL_template_syntax
178 
179  A list of template parameters currently supported and what they are replaced with:
180  \table
181  \header \o parameter
182  \o value
183  \row \o "{count}"
184  \o "20"
185  \row \o "{startIndex}"
186  \o "0"
187  \row \o "{startPage}"
188  \o "0"
189  \row \o "{language}"
190  \o "the default language code (RFC 3066)"
191  \row \o "{inputEncoding}"
192  \o "UTF-8"
193  \row \o "{outputEncoding}"
194  \o "UTF-8"
195  \row \o "{*:source}"
196  \o "application name, QCoreApplication::applicationName()"
197  \row \o "{searchTerms}"
198  \o "the string supplied by the user"
199  \endtable
200 
201  \sa searchUrlTemplate(), searchParameters(), suggestionsUrl()
202 */
203 QUrl OpenSearchEngine::searchUrl(const QString &searchTerm) const
204 {
205  if (m_searchUrlTemplate.isEmpty())
206  return QUrl();
207 
208  QUrl retVal = QUrl::fromEncoded(parseTemplate(searchTerm, m_searchUrlTemplate).toUtf8());
209  QUrlQuery query(retVal.query());
210 
211  if (m_searchMethod != QLatin1String("post")) {
212  Parameters::const_iterator end = m_searchParameters.constEnd();
213  Parameters::const_iterator i = m_searchParameters.constBegin();
214  for (; i != end; ++i)
215  query.addQueryItem(i->first, parseTemplate(searchTerm, i->second));
216  }
217  retVal.setQuery(query);
218 
219  return retVal;
220 }

452 /*!
453  Requests contextual suggestions on the search engine, for a given \a searchTerm.
454 
455  If succeeded, suggestions() signal will be emitted once the suggestions are received.
456 
457  \note To be able to request suggestions, you need to provide a network access manager,
458  which will be used for network operations.
459 
460  \sa requestSearchResults()
461 */
462 void OpenSearchEngine::requestSuggestions(const QString &searchTerm)
463 {
464  if (searchTerm.isEmpty() || !providesSuggestions())
465  return;
466 
467  Q_ASSERT(m_networkAccessManager);
468 
469  if (!m_networkAccessManager)
470  return;
471 
472  if (m_suggestionsReply) {
473  m_suggestionsReply->disconnect(this);
474  m_suggestionsReply->abort();
475  m_suggestionsReply->deleteLater();
476  m_suggestionsReply = nullptr;
477  }
478 
479  Q_ASSERT(m_requestMethods.contains(m_suggestionsMethod));
480  if (m_suggestionsMethod == QLatin1String("get")) {
481  m_suggestionsReply = m_networkAccessManager->get(QNetworkRequest(suggestionsUrl(searchTerm)));
482  } else {
483  QStringList parameters;
484  Parameters::const_iterator end = m_suggestionsParameters.constEnd();
485  Parameters::const_iterator i = m_suggestionsParameters.constBegin();
486  for (; i != end; ++i)
487  parameters.append(i->first + QLatin1String("=") + i->second);
488 
489  QByteArray data = parameters.join(QLatin1String("&")).toUtf8();
490  m_suggestionsReply = m_networkAccessManager->post(QNetworkRequest(suggestionsUrl(searchTerm)), data);
491  }
492 
493  connect(m_suggestionsReply, SIGNAL(finished()), this, SLOT(suggestionsObtained()));
494 }

496 /*!
497  Requests search results on the search engine, for a given \a searchTerm.
498 
499  The default implementation does nothing, to supply your own you need to create your own
500  OpenSearchEngineDelegate subclass and supply it to the engine. Then the function will call
501  the performSearchRequest() method of the delegate, which can then handle the request
502  in a custom way.
503 
504  \sa requestSuggestions(), delegate()
505 */
506 void OpenSearchEngine::requestSearchResults(const QString &searchTerm)
507 {
508  if (!m_delegate || searchTerm.isEmpty())
509  return;
510 
511  Q_ASSERT(m_requestMethods.contains(m_searchMethod));
512 
513  QNetworkRequest request(QUrl(searchUrl(searchTerm)));
514  QByteArray data;
515  QNetworkAccessManager::Operation operation = m_requestMethods.value(m_searchMethod);
516 
517  if (operation == QNetworkAccessManager::PostOperation) {
518  QStringList parameters;
519  Parameters::const_iterator end = m_searchParameters.constEnd();
520  Parameters::const_iterator i = m_searchParameters.constBegin();
521  for (; i != end; ++i)
522  parameters.append(i->first + QLatin1String("=") + i->second);
523 
524  data = parameters.join(QLatin1String("&")).toUtf8();
525  }
526 
527  m_delegate->performSearchRequest(request, operation, data);
528 }

530 void OpenSearchEngine::suggestionsObtained()
531 {
532  QString response(QString::fromUtf8(m_suggestionsReply->readAll()));
533  response = response.trimmed();
534 
535  m_suggestionsReply->close();
536  m_suggestionsReply->deleteLater();
537  m_suggestionsReply = nullptr;
538 
539  if (response.isEmpty())
540  return;
541 
542  if (!response.startsWith(QLatin1Char('[')) || !response.endsWith(QLatin1Char(']')))
543  return;
544 
545  if (!m_scriptEngine)
546  m_scriptEngine = new QScriptEngine();
547 
548  // Evaluate the JSON response using QtScript.
549  if (!m_scriptEngine->canEvaluate(response))
550  return;
551 
552  QScriptValue responseParts = m_scriptEngine->evaluate(response);
553 
554  if (!responseParts.property(1).isArray())
555  return;
556 
557  QStringList suggestionsList;
558  qScriptValueToSequence(responseParts.property(1), suggestionsList);
559 
560  emit suggestions(suggestionsList);
561 }
