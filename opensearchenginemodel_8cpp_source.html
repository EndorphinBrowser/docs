85 QVariant OpenSearchEngineModel::data(const QModelIndex &index, int role) const
86 {
87  if (index.row() >= m_manager->enginesCount() || index.row() < 0)
88  return QVariant();
89 
90  OpenSearchEngine *engine = m_manager->engine(m_manager->allEnginesNames().at(index.row()));
91 
92  if (!engine)
93  return QVariant();
94 
95  switch (index.column()) {
96  case 0:
97  switch (role) {
98  case Qt::DisplayRole:
99  return engine->name();
100  break;
101  case Qt::DecorationRole: {
102  QImage image = engine->image();
103  if (image.isNull())
104  return BrowserApplication::icon(engine->imageUrl());
105  return image;
106  break;
107  }
108  case Qt::ToolTipRole:
109  QString description = tr("<strong>Description:</strong> %1").arg(engine->description());
110 
111  if (engine->providesSuggestions()) {
112  description += QLatin1String("<br />");
113  description += tr("<strong>Provides contextual suggestions</strong>");
114  }
115 
116  return description;
117  break;
118  }
119  break;
120 
121  case 1:
122  switch (role) {
123  case Qt::EditRole:
124  case Qt::DisplayRole:
125  return QStringList(m_manager->keywordsForEngine(engine)).join(QLatin1String(","));
126  case Qt::ToolTipRole:
127  return tr("Comma-separated list of keywords that may be entered in the location bar"
128  "followed by search terms to search with this engine");
129  }
130  break;
131  }
132 
133  return QVariant();
134 }
