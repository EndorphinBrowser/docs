26 /*!
27  \class OpenSearchReader
28  \brief A class reading a search engine description from an external source
29 
30  OpenSearchReader is a class that can be used to read search engine descriptions
31  formed using the OpenSearch format.
32 
33  It inherits QXmlStreamReader and thus provides additional functions, such as
34  QXmlStreamReader::error(), QXmlStreamReader::hasError() that can be used to make sure
35  the reading procedure succeeded.

37  For more information see:
38  http://www.opensearch.org/Specifications/OpenSearch/1.1/Draft_4#OpenSearch_description_document
39 
40  \sa OpenSearchEngine, OpenSearchWriter

53 /*!

80 OpenSearchEngine *OpenSearchReader::read()
81 {
82  OpenSearchEngine *engine = new OpenSearchEngine();
83 
84  while (!isStartElement() && !atEnd())
85  readNext();
86 
87  if (name() != QLatin1String("OpenSearchDescription")
88  || namespaceUri() != QLatin1String("http://a9.com/-/spec/opensearch/1.1/")) {
89  raiseError(QObject::tr("The file is not an OpenSearch 1.1 file."));
90  return engine;
91  }
92 
93  while (!atEnd()) {
94  readNext();
95 
96  if (!isStartElement())
97  continue;
98 
99  if (name() == QLatin1String("ShortName")) {
100  engine->setName(readElementText());
101  } else if (name() == QLatin1String("Description")) {
102  engine->setDescription(readElementText());
103  } else if (name() == QLatin1String("Url")) {
104 
105  QString type = attributes().value(QLatin1String("type")).toString();
106  QString url = attributes().value(QLatin1String("template")).toString();
107  QString method = attributes().value(QLatin1String("method")).toString();
108 
109  if (type == QLatin1String("application/x-suggestions+json")
110  && !engine->suggestionsUrlTemplate().isEmpty())
111  continue;
112 
113  if ((type.isEmpty()
114  || type == QLatin1String("text/html")
115  || type == QLatin1String("application/xhtml+xml"))
116  && !engine->searchUrlTemplate().isEmpty())
117  continue;
118 
119  if (url.isEmpty())
120  continue;
121 
122  QList<OpenSearchEngine::Parameter> parameters;
123 
124  readNext();
125 
126  while (!(isEndElement() && name() == QLatin1String("Url"))) {
127  if (!isStartElement() || (name() != QLatin1String("Param") && name() != QLatin1String("Parameter"))) {
128  readNext();
129  continue;
130  }
131 
132  QString key = attributes().value(QLatin1String("name")).toString();
133  QString value = attributes().value(QLatin1String("value")).toString();
134 
135  if (!key.isEmpty() && !value.isEmpty())
136  parameters.append(OpenSearchEngine::Parameter(key, value));
137 
138  while (!isEndElement())
139  readNext();
140  }
141 
142  if (type == QLatin1String("application/x-suggestions+json")) {
143  engine->setSuggestionsUrlTemplate(url);
144  engine->setSuggestionsParameters(parameters);
145  engine->setSuggestionsMethod(method);
146  } else if (type.isEmpty() || type == QLatin1String("text/html") || type == QLatin1String("application/xhtml+xml")) {
147  engine->setSearchUrlTemplate(url);
148  engine->setSearchParameters(parameters);
149  engine->setSearchMethod(method);
150  }
151 
152  } else if (name() == QLatin1String("Image")) {
153  engine->setImageUrl(readElementText());
154  }
155 
156  if (!engine->name().isEmpty()
157  && !engine->description().isEmpty()
158  && !engine->suggestionsUrlTemplate().isEmpty()
159  && !engine->searchUrlTemplate().isEmpty()
160  && !engine->imageUrl().isEmpty())
161  break;
162  }
163 
164  return engine;
165 }
