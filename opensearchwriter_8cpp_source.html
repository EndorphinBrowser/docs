74 void OpenSearchWriter::write(OpenSearchEngine *engine)
75 {
76  writeStartDocument();
77  writeStartElement(QLatin1String("OpenSearchDescription"));
78  writeDefaultNamespace(QLatin1String("http://a9.com/-/spec/opensearch/1.1/"));
79 
80  if (!engine->name().isEmpty())
81  writeTextElement(QLatin1String("ShortName"), engine->name());
82 
83  if (!engine->description().isEmpty())
84  writeTextElement(QLatin1String("Description"), engine->description());
85 
86  if (!engine->searchUrlTemplate().isEmpty()) {
87  writeStartElement(QLatin1String("Url"));
88  writeAttribute(QLatin1String("method"), engine->searchMethod());
89  writeAttribute(QLatin1String("type"), QLatin1String("text/html"));
90  writeAttribute(QLatin1String("template"), engine->searchUrlTemplate());
91 
92  if (!engine->searchParameters().empty()) {
93  writeNamespace(QLatin1String("http://a9.com/-/spec/opensearch/extensions/parameters/1.0/"), QLatin1String("p"));
94 
95  QList<OpenSearchEngine::Parameter>::const_iterator end = engine->searchParameters().constEnd();
96  QList<OpenSearchEngine::Parameter>::const_iterator i = engine->searchParameters().constBegin();
97  for (; i != end; ++i) {
98  writeStartElement(QLatin1String("p:Parameter"));
99  writeAttribute(QLatin1String("name"), i->first);
100  writeAttribute(QLatin1String("value"), i->second);
101  writeEndElement();
102  }
103  }
104 
105  writeEndElement();
106  }
107 
108  if (!engine->suggestionsUrlTemplate().isEmpty()) {
109  writeStartElement(QLatin1String("Url"));
110  writeAttribute(QLatin1String("method"), engine->suggestionsMethod());
111  writeAttribute(QLatin1String("type"), QLatin1String("application/x-suggestions+json"));
112  writeAttribute(QLatin1String("template"), engine->suggestionsUrlTemplate());
113 
114  if (!engine->suggestionsParameters().empty()) {
115  writeNamespace(QLatin1String("http://a9.com/-/spec/opensearch/extensions/parameters/1.0/"), QLatin1String("p"));
116 
117  QList<OpenSearchEngine::Parameter>::const_iterator end = engine->suggestionsParameters().constEnd();
118  QList<OpenSearchEngine::Parameter>::const_iterator i = engine->suggestionsParameters().constBegin();
119  for (; i != end; ++i) {
120  writeStartElement(QLatin1String("p:Parameter"));
121  writeAttribute(QLatin1String("name"), i->first);
122  writeAttribute(QLatin1String("value"), i->second);
123  writeEndElement();
124  }
125  }
126 
127  writeEndElement();
128  }
129 
130  if (!engine->imageUrl().isEmpty())
131  writeTextElement(QLatin1String("Image"), engine->imageUrl());
132 
133  writeEndElement();
134  writeEndDocument();
135 }
