42 void PlainTextEditSearch::find(QTextDocument::FindFlags flags)
43 {
44  QString searchString = ui.searchLineEdit->text();
45  if (!m_edit || searchString.isEmpty())
46  return;
47  if (searchString != m_lastSearchTerm) {
48  QTextCursor cursor = m_edit->textCursor();
49  cursor.setPosition(cursor.selectionStart());
50  cursor.clearSelection();
51  m_edit->setTextCursor(cursor);
52  m_lastSearchTerm = searchString;
53  }
54  QString infoString;
55  if (!m_edit->find(searchString, flags)) {
56  /* no support for wrapping so we set the cursor to start */
57  QTextCursor cursor = m_edit->textCursor();
58  m_edit->moveCursor(QTextCursor::Start);
59  /* ...search again from the beginning */
60  if (!m_edit->find(searchString,flags)) {
61  /* and if there's still nothing, we revert the cursor */
62  infoString = tr("Not Found");
63  cursor.clearSelection();
64  m_edit->setTextCursor(cursor);
65  }
66  }
67  ui.searchInfo->setText(infoString);
68 }
