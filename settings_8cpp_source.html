98 void SettingsDialog::loadDefaults()
99 {
100  QWebEngineSettings *defaultSettings = QWebEngineSettings::globalSettings();
101  QString standardFontFamily = defaultSettings->fontFamily(QWebEngineSettings::StandardFont);
102  int standardFontSize = defaultSettings->fontSize(QWebEngineSettings::DefaultFontSize);
103  m_standardFont = QFont(standardFontFamily, standardFontSize);
104  standardLabel->setText(QString(QLatin1String("%1 %2")).arg(m_standardFont.family()).arg(m_standardFont.pointSize()));
105 
106  QString fixedFontFamily = defaultSettings->fontFamily(QWebEngineSettings::FixedFont);
107  int fixedFontSize = defaultSettings->fontSize(QWebEngineSettings::DefaultFixedFontSize);
108  m_fixedFont = QFont(fixedFontFamily, fixedFontSize);
109  fixedLabel->setText(QString(QLatin1String("%1 %2")).arg(m_fixedFont.family()).arg(m_fixedFont.pointSize()));
110 
111  downloadsLocation->setText(QStandardPaths::writableLocation(QStandardPaths::DesktopLocation));
112 
113  blockPopupWindows->setChecked(!defaultSettings->testAttribute(QWebEngineSettings::JavascriptCanOpenWindows));
114  enableJavascript->setChecked(defaultSettings->testAttribute(QWebEngineSettings::JavascriptEnabled));
115  enableImages->setChecked(defaultSettings->testAttribute(QWebEngineSettings::AutoLoadImages));
116  enableLocalStorage->setChecked(defaultSettings->testAttribute(QWebEngineSettings::LocalStorageEnabled));
117 
118  minimFontSizeCheckBox->setChecked(false);
119  minimumFontSizeSpinBox->setValue(9);
120 }

122 void SettingsDialog::loadFromSettings()
123 {
124  QSettings settings;
125  settings.beginGroup(QLatin1String("Settings"));
126  tabWidget->setCurrentIndex(settings.value(QLatin1String("currentTab"), 0).toInt());
127  settings.endGroup();
128 
129  settings.beginGroup(QLatin1String("MainWindow"));
130  QString defaultHome = QLatin1String("about:home");
131  homeLineEdit->setText(settings.value(QLatin1String("home"), defaultHome).toString());
132  startupBehavior->setCurrentIndex(settings.value(QLatin1String("startupBehavior"), 0).toInt());
133  settings.endGroup();
134 
135  settings.beginGroup(QLatin1String("history"));
136  int historyExpire = settings.value(QLatin1String("historyLimit")).toInt();
137  int idx = 0;
138  switch (historyExpire) {
139  case 1:
140  idx = 0;
141  break;
142  case 7:
143  idx = 1;
144  break;
145  case 14:
146  idx = 2;
147  break;
148  case 30:
149  idx = 3;
150  break;
151  case 365:
152  idx = 4;
153  break;
154  case -1:
155  idx = 5;
156  break;
157  case -2:
158  idx = 6;
159  break;
160  default:
161  idx = 5;
162  }
163  expireHistory->setCurrentIndex(idx);
164  settings.endGroup();
165 
166  settings.beginGroup(QLatin1String("urlloading"));
167  bool search = settings.value(QLatin1String("searchEngineFallback"), false).toBool();
168  searchEngineFallback->setChecked(search);
169  settings.endGroup();
170 
171  settings.beginGroup(QLatin1String("downloadmanager"));
172  bool alwaysPromptForFileName = settings.value(QLatin1String("alwaysPromptForFileName"), false).toBool();
173  downloadAsk->setChecked(alwaysPromptForFileName);
174  QString downloadDirectory = settings.value(QLatin1String("downloadDirectory"), downloadsLocation->text()).toString();
175  downloadsLocation->setText(downloadDirectory);
176  externalDownloadButton->setChecked(settings.value(QLatin1String("external"), false).toBool());
177  externalDownloadPath->setText(settings.value(QLatin1String("externalPath")).toString());
178  settings.endGroup();
179 
180  // Appearance
181  settings.beginGroup(QLatin1String("websettings"));
182  m_fixedFont = settings.value(QLatin1String("fixedFont"), m_fixedFont).value<QFont>();
183  m_standardFont = settings.value(QLatin1String("standardFont"), m_standardFont).value<QFont>();
184 
185  standardLabel->setText(QString(QLatin1String("%1 %2")).arg(m_standardFont.family()).arg(m_standardFont.pointSize()));
186  fixedLabel->setText(QString(QLatin1String("%1 %2")).arg(m_fixedFont.family()).arg(m_fixedFont.pointSize()));
187 
188  blockPopupWindows->setChecked(settings.value(QLatin1String("blockPopupWindows"), blockPopupWindows->isChecked()).toBool());
189  enableJavascript->setChecked(settings.value(QLatin1String("enableJavascript"), enableJavascript->isChecked()).toBool());
190  enableImages->setChecked(settings.value(QLatin1String("enableImages"), enableImages->isChecked()).toBool());
191  enableLocalStorage->setChecked(settings.value(QLatin1String("enableLocalStorage"), enableLocalStorage->isChecked()).toBool());
192  userStyleSheet->setText(QString::fromUtf8(settings.value(QLatin1String("userStyleSheet")).toUrl().toEncoded()));
193  int minimumFontSize = settings.value(QLatin1String("minimumFontSize"), 0).toInt();
194  minimFontSizeCheckBox->setChecked(minimumFontSize != 0);
195  if (minimumFontSize != 0)
196  minimumFontSizeSpinBox->setValue(minimumFontSize);
197  settings.endGroup();
198 
199  // Network
200  settings.beginGroup(QLatin1String("network"));
201  m_cacheEnabled = settings.value(QLatin1String("cacheEnabled"), true).toBool();
202  networkCache->setChecked(m_cacheEnabled);
203  networkCacheMaximumSizeSpinBox->setValue(settings.value(QLatin1String("maximumCacheSize"), 50).toInt());
204  settings.endGroup();
205 
206  // Proxy
207  settings.beginGroup(QLatin1String("proxy"));
208  proxySupport->setChecked(settings.value(QLatin1String("enabled"), false).toBool());
209  proxyType->setCurrentIndex(settings.value(QLatin1String("type"), 0).toInt());
210  proxyHostName->setText(settings.value(QLatin1String("hostName")).toString());
211  proxyPort->setValue(settings.value(QLatin1String("port"), 1080).toInt());
212  proxyUserName->setText(settings.value(QLatin1String("userName")).toString());
213  proxyPassword->setText(settings.value(QLatin1String("password")).toString());
214  settings.endGroup();
215 
216  // Tabs
217  settings.beginGroup(QLatin1String("tabs"));
218  selectTabsWhenCreated->setChecked(settings.value(QLatin1String("selectNewTabs"), false).toBool());
219  confirmClosingMultipleTabs->setChecked(settings.value(QLatin1String("confirmClosingMultipleTabs"), true).toBool());
220  oneCloseButton->setChecked(settings.value(QLatin1String("oneCloseButton"),false).toBool());
221  quitAsLastTabClosed->setChecked(settings.value(QLatin1String("quitAsLastTabClosed"), true).toBool());
222  openTargetBlankLinksIn->setCurrentIndex(settings.value(QLatin1String("openTargetBlankLinksIn"), TabWidget::NewSelectedTab).toInt());
223  openLinksFromAppsIn->setCurrentIndex(settings.value(QLatin1String("openLinksFromAppsIn"), TabWidget::NewSelectedTab).toInt());
224  settings.endGroup();
225 
226  // Accessibility
227  settings.beginGroup(QLatin1String("WebView"));
228  enableAccessKeys->setChecked(settings.value(QLatin1String("enableAccessKeys"), true).toBool());
229  settings.endGroup();
230 }

232 void SettingsDialog::saveToSettings()
233 {
234  QSettings settings;
235  settings.beginGroup(QLatin1String("Settings"));
236  settings.setValue(QLatin1String("currentTab"), tabWidget->currentIndex());
237  settings.endGroup();

242  settings.endGroup();
243 
244  settings.beginGroup(QLatin1String("downloadmanager"));
245  settings.setValue(QLatin1String("alwaysPromptForFileName"), downloadAsk->isChecked());
246  settings.setValue(QLatin1String("downloadDirectory"), downloadsLocation->text());
247  settings.setValue(QLatin1String("external"), externalDownloadButton->isChecked());
248  settings.setValue(QLatin1String("externalPath"), externalDownloadPath->text());
249  settings.endGroup();
250 
251  settings.beginGroup(QLatin1String("history"));
252  int historyExpire = expireHistory->currentIndex();
253  int idx = -1;
254  switch (historyExpire) {
255  case 0:
256  idx = 1;
257  break;
258  case 1:
259  idx = 7;
260  break;
261  case 2:
262  idx = 14;
263  break;
264  case 3:
265  idx = 30;
266  break;
267  case 4:
268  idx = 365;
269  break;
270  case 5:
271  idx = -1;
272  break;
273  case 6:
274  idx = -2;
275  break;
276  }
277  settings.setValue(QLatin1String("historyLimit"), idx);
278  settings.endGroup();
279 
280  settings.beginGroup(QLatin1String("urlloading"));
281  settings.setValue(QLatin1String("searchEngineFallback"), searchEngineFallback->isChecked());
282  settings.endGroup();
283 
284  // Appearance
285  settings.beginGroup(QLatin1String("websettings"));
286  settings.setValue(QLatin1String("fixedFont"), m_fixedFont);
287  settings.setValue(QLatin1String("standardFont"), m_standardFont);
288 
289  settings.setValue(QLatin1String("blockPopupWindows"), blockPopupWindows->isChecked());
290  settings.setValue(QLatin1String("enableJavascript"), enableJavascript->isChecked());
291  settings.setValue(QLatin1String("enableImages"), enableImages->isChecked());
292  settings.setValue(QLatin1String("enableLocalStorage"), enableLocalStorage->isChecked());
293  QString userStyleSheetString = userStyleSheet->text();
294  if (QFile::exists(userStyleSheetString))
295  settings.setValue(QLatin1String("userStyleSheet"), QUrl::fromLocalFile(userStyleSheetString));
296  else
297  settings.setValue(QLatin1String("userStyleSheet"), QUrl::fromEncoded(userStyleSheetString.toUtf8()));
298 
299  if (minimFontSizeCheckBox->isChecked())
300  settings.setValue(QLatin1String("minimumFontSize"), minimumFontSizeSpinBox->value());
301  else
302  settings.setValue(QLatin1String("minimumFontSize"), 0);
303  settings.endGroup();
304 
305  // Network
306  settings.beginGroup(QLatin1String("network"));
307  settings.setValue(QLatin1String("cacheEnabled"), networkCache->isChecked());
308  settings.setValue(QLatin1String("maximumCacheSize"), networkCacheMaximumSizeSpinBox->value());
309  settings.endGroup();
310 
311  // proxy

313  settings.setValue(QLatin1String("enabled"), proxySupport->isChecked());
314  settings.setValue(QLatin1String("type"), proxyType->currentIndex());
315  settings.setValue(QLatin1String("hostName"), proxyHostName->text());
316  settings.setValue(QLatin1String("port"), proxyPort->text());
317  settings.setValue(QLatin1String("userName"), proxyUserName->text());
318  settings.setValue(QLatin1String("password"), proxyPassword->text());
319  settings.endGroup();
320 
321  // Tabs
322  settings.beginGroup(QLatin1String("tabs"));
323  settings.setValue(QLatin1String("selectNewTabs"), selectTabsWhenCreated->isChecked());
324  settings.setValue(QLatin1String("confirmClosingMultipleTabs"), confirmClosingMultipleTabs->isChecked());
325  settings.setValue(QLatin1String("oneCloseButton"), oneCloseButton->isChecked());
326  settings.setValue(QLatin1String("quitAsLastTabClosed"), quitAsLastTabClosed->isChecked());
327  settings.setValue(QLatin1String("openTargetBlankLinksIn"), openTargetBlankLinksIn->currentIndex());
328  settings.setValue(QLatin1String("openLinksFromAppsIn"), openLinksFromAppsIn->currentIndex());
329  settings.endGroup();
330 
331  // Accessibility

333  settings.setValue(QLatin1String("enableAccessKeys"), enableAccessKeys->isChecked());
334  settings.endGroup();

336  BrowserApplication::instance()->loadSettings();
