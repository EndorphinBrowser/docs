51 bool SingleApplication::sendMessage(const QByteArray &message, int waitMsecsForReply)
52 {
53 #ifdef SINGALAPPLICATION_DEBUG
54  qDebug() << "SingleApplication::" << __FUNCTION__ << message << waitMsecsForReply;
55 #endif
56  QLocalSocket socket;
57  socket.connectToServer(serverName());
58  if (!socket.waitForConnected(500))
59  return false;
60  socket.write(message);
61  socket.flush();
62  socket.waitForBytesWritten();
63  bool success = true;
64  if (socket.error() != QLocalSocket::UnknownSocketError) {
65 #ifdef SINGALAPPLICATION_DEBUG
66  qDebug() << "SingleApplication::" << __FUNCTION__ << socket.errorString();
67 #endif
68  success = false;
69  }
70  if (success) {
71  socket.waitForReadyRead(waitMsecsForReply);
72  if (socket.bytesAvailable() > 0)
73  emit messageReceived(&socket);
74  }
75  return success;
76 }

78 bool SingleApplication::startSingleServer()
79 {
80  if (m_localServer)
81  return false;
82 
83  m_localServer = new QLocalServer(this);
84  connect(m_localServer, SIGNAL(newConnection()),
85  this, SLOT(newConnection()));
86  bool success = false;
87  if (!m_localServer->listen(serverName())) {
88  if (QAbstractSocket::AddressInUseError == m_localServer->serverError()) {
89  // cleanup from a segfaulted server
90 #ifdef Q_OS_UNIX
91  QString fullServerName = QDir::tempPath() + QLatin1String("/") + serverName();
92  if (QFile::exists(fullServerName))
93  QFile::remove(fullServerName);
94 #endif
95  if (m_localServer->listen(serverName())) {
96  success = true;
97  }
98  }
99  if (!success) {
100  qWarning() << "SingleApplication: Unable to listen:" << m_localServer->errorString();
101  }
102  } else {
103  success = true;
104  }
105 
106  if (success) {
107 #ifdef Q_OS_UNIX
108  QFile file(m_localServer->fullServerName());
109  if (!file.setPermissions(QFile::ReadUser | QFile::WriteUser))
110  qWarning() << "SingleApplication: Unable to set permissions on:"
111  << file.fileName() << file.errorString();
112 #endif
113  }
114 
115  if (!success) {
116  delete m_localServer;
117  m_localServer = nullptr;
118  }
119  return success;
120 }

137 QString SingleApplication::serverName() const
138 {
139  QString serverName = QCoreApplication::applicationName();
140  Q_ASSERT(!serverName.isEmpty());
141 #ifdef Q_WS_QWS
142  serverName += QLatin1String("_qws");
143 #endif
144 #ifndef Q_OS_WIN
145  serverName += QString(QLatin1String("_%1_%2")).arg(getuid()).arg(getgid());
146 #else
147  static QString login;
148  if (login.isEmpty()) {
149  QT_WA({
150  wchar_t buffer[256];
151  DWORD bufferSize = sizeof(buffer) / sizeof(wchar_t) - 1;
152  GetUserNameW(buffer, &bufferSize);
153  login = QString::fromUtf16((ushort*)buffer);
154  },
155  {
156  char buffer[256];
157  DWORD bufferSize = sizeof(buffer) / sizeof(char) - 1;
158  GetUserNameA(buffer, &bufferSize);
159  login = QString::fromLocal8Bit(buffer);
160  });
161  }
162  serverName += QString::fromAscii("_%1").arg(login);
163 #endif
164  return serverName;
165 }
