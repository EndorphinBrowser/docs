97 void SourceHighlighter::highlightBlock(const QString &text)
98 {
99  int state = previousBlockState();
100  int len = text.length();
101  int start = 0;
102  int pos = 0;
103  QRegExp regex;
104 
105  while (pos >= 0 && pos < len && len > 0) {
106  switch (state) {
107  default:
108  case Normal:
109  regex.setPattern(QLatin1String("[<&]"));
110  pos = regex.indexIn(text, pos);
111  if (pos >= 0) {
112  if (text.at(pos) == QLatin1Char('<')) {
113  start = pos;
114  if (text.mid(pos, 4) == QLatin1String("<!--")) {
115  state = InComment;
116  } else {
117  state = InTag;
118  }
119  ++pos;
120  } else if (text.at(pos) == QLatin1Char('&')) {
121  regex.setPattern(QLatin1String("&[a-zA-Z0-9]+;"));
122  if (regex.indexIn(text, pos) == pos) {
123  setFormat(pos, regex.matchedLength(), formats[Entity]);
124  }
125  ++pos;
126  }
127  }
128  break;
129  case InComment:
130  regex.setPattern(QLatin1String("-->"));
131  pos = regex.indexIn(text, pos);
132  if (pos >= 0) {
133  state = Normal;
134  pos += 3;
135  setFormat(start, pos - start, formats[Comment]);
136  ++pos;
137  } else {
138  setFormat(start, len - start, formats[Comment]);
139  }
140  break;
141  case InTag:
142  regex.setPattern(QLatin1String("[>\"]"));
143  pos = regex.indexIn(text, pos);
144  if (pos >= 0) {
145  if (text.at(pos) == QLatin1Char('>')) {
146  state = Normal;
147  ++pos;
148  setFormat(start, pos - start, formats[Tag]);
149  } else if (text.at(pos) == QLatin1Char('"')) {
150  setFormat(start, pos - start, formats[Tag]);
151  start = pos;
152  state = InAttribute;
153  ++pos;
154  }
155  } else {
156  setFormat(start, len-start, formats[Tag]);
157  }
158  break;
159  case InAttribute:
160  regex.setPattern(QLatin1String("\""));
161  pos = regex.indexIn(text, pos);
162  if (pos >= 0) {
163  setFormat(start, pos - start, formats[Attribute]);
164  state = InTag;
165  start = ++pos;
166  } else {
167  setFormat(start, len - start, formats[Attribute]);
168  }
169  break;
170  }
171  }
172  setCurrentBlockState(state);
173 }
