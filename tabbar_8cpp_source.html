99  connect(this, SIGNAL(customContextMenuRequested(const QPoint &)),
100  this, SLOT(contextMenuRequested(const QPoint &)));
101 
102  QString alt = QLatin1String("Ctrl+%1");
103  for (int i = 0; i < 9; ++i) {

109  m_viewTabBarAction = new QAction(this);
110  updateViewToolBarAction();
111  connect(m_viewTabBarAction, SIGNAL(triggered()),

160 void TabBar::contextMenuRequested(const QPoint &position)
161 {
162  QMenu menu;
163  TabWidget *tabWidget = qobject_cast<TabWidget*>(parentWidget());
164  if (!tabWidget)
165  return;
166 
167  menu.addAction(tabWidget->newTabAction());
168  int index = tabAt(position);
169  if (-1 != index) {
170  QAction *action = menu.addAction(tr("Duplicate Tab"),
171  this, SLOT(cloneTab()));
172  action->setData(index);
173 
174  menu.addSeparator();
175 
176  action = menu.addAction(tr("&Close Tab"),
177  this, SLOT(closeTab()), QKeySequence::Close);
178  action->setData(index);
179 
180  action = menu.addAction(tr("Close &Other Tabs"),
181  this, SLOT(closeOtherTabs()));
182  action->setData(index);
183 
184  menu.addSeparator();
185 
186  action = menu.addAction(tr("Reload Tab"),
187  this, SLOT(reloadTab()), QKeySequence::Refresh);
188  action->setData(index);
189  } else {
190  menu.addSeparator();
191  }
192  menu.addAction(tr("Reload All Tabs"), this, SIGNAL(reloadAllTabs()));
193  menu.addSeparator();
194  menu.addAction(tabWidget->bookmarkTabsAction());
195  menu.exec(QCursor::pos());
196 }

256 void TabBar::mouseMoveEvent(QMouseEvent *event)
257 {
258  if (event->buttons() == Qt::LeftButton) {
259  int diffX = event->pos().x() - m_dragStartPos.x();
260  int diffY = event->pos().y() - m_dragStartPos.y();
261  if ((event->pos() - m_dragStartPos).manhattanLength() > QApplication::startDragDistance()
262  && diffX < 3 && diffX > -3
263  && diffY < -10) {
264  QDrag *drag = new QDrag(this);
265  QMimeData *mimeData = new QMimeData;
266  QList<QUrl> urls;
267  int index = tabAt(event->pos());
268  QUrl url = tabData(index).toUrl();
269  urls.append(url);
270  mimeData->setUrls(urls);
271  mimeData->setText(tabText(index));
272  mimeData->setData(QLatin1String("action"), "tab-reordering");
273  drag->setMimeData(mimeData);
274  drag->exec();
275  }
276  }
277  QTabBar::mouseMoveEvent(event);
278 }

289 void TabBar::dropEvent(QDropEvent *event)
290 {
291  const QMimeData *mimeData = event->mimeData();
292  QUrl url;
293  if (mimeData->hasUrls())
294  url = mimeData->urls().at(0);
295  else if (mimeData->hasText())
296  url = QUrl::fromEncoded(mimeData->text().toUtf8());
297 
298  if (!url.isEmpty() && url.isValid()) {
299  event->acceptProposedAction();
300  int index = tabAt(event->pos());
301  if (-1 != index) {
302  setCurrentIndex(index);
303  emit loadUrl(url, TabWidget::CurrentTab);
304  } else {
305  emit loadUrl(url, TabWidget::NewSelectedTab);
306  }
307  }
308 
309  QTabBar::dropEvent(event);
310 }
