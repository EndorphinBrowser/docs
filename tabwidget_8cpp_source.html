243 void TabWidget::currentChanged(int index)
244 {
245  WebView *webView = this->webView(index);
246  if (!webView)
247  return;
248 
249  Q_ASSERT(m_locationBars->count() == count());
250 
251  WebView *oldWebView = this->webView(m_locationBars->currentIndex());
252  if (oldWebView) {
253  disconnect(oldWebView->page(), SIGNAL(linkHovered(const QString&)),
254  this, SIGNAL(linkHovered(const QString&)));
255  disconnect(oldWebView, SIGNAL(loadProgress(int)),
256  this, SIGNAL(loadProgress(int)));
257  }
258 
259  connect(webView->page(), SIGNAL(linkHovered(const QString&)),
260  this, SIGNAL(linkHovered(const QString&)));
261  connect(webView, SIGNAL(loadProgress(int)),
262  this, SIGNAL(loadProgress(int)));
263 
264  for (int i = 0; i < m_actions.count(); ++i) {
265  WebActionMapper *mapper = m_actions[i];
266  mapper->updateCurrent(webView->page());
267  }
268  emit setCurrentTitle(webView->title());
269  m_locationBars->setCurrentIndex(index);
270  emit loadProgress(webView->progress());
271  emit showStatusBarMessage(webView->lastStatusBarText());
272  if (webView->url().isEmpty() && webView->hasFocus()) {
273  m_locationBars->currentWidget()->setFocus();
274  } else if (!webView->url().isEmpty()) {
275  webView->setFocus();
276  }
277 }

364 WebView *TabWidget::makeNewTab(bool makeCurrent)
365 {
366  // line edit
367  LocationBar *locationBar = new LocationBar;
368  if (!m_lineEditCompleter) {
369  HistoryCompletionModel *completionModel = new HistoryCompletionModel(this);
370  completionModel->setSourceModel(BrowserApplication::historyManager()->historyFilterModel());
371  m_lineEditCompleter = new HistoryCompleter(completionModel, this);
372  connect(m_lineEditCompleter, SIGNAL(activated(const QString &)),
373  this, SLOT(loadString(const QString &)));
374  // Should this be in Qt by default?
375  QAbstractItemView *popup = m_lineEditCompleter->popup();
376  QListView *listView = qobject_cast<QListView*>(popup);
377  if (listView) {
378  // Urls are always LeftToRight
379  listView->setLayoutDirection(Qt::LeftToRight);
380  listView->setUniformItemSizes(true);
381  }
382  }
383  locationBar->setCompleter(m_lineEditCompleter);
384  connect(locationBar, SIGNAL(returnPressed()), this, SLOT(lineEditReturnPressed()));
385  m_locationBars->addWidget(locationBar);
386  m_locationBars->setSizePolicy(locationBar->sizePolicy());
387 
388 #ifndef AUTOTESTS
389  QWidget::setTabOrder(locationBar, BrowserMainWindow::parentWindow(this)->findChild<ToolbarSearch*>());
390 #endif
391 
392  // webview
393  WebView *webView = new WebView;
394  locationBar->setWebView(webView);
395  connect(webView, SIGNAL(loadStarted()),
396  this, SLOT(webViewLoadStarted()));
397  connect(webView, SIGNAL(loadProgress(int)),
398  this, SLOT(webViewLoadProgress(int)));
399  connect(webView, SIGNAL(loadFinished(bool)),
400  this, SLOT(webViewLoadFinished(bool)));
401  connect(webView, SIGNAL(iconChanged(QIcon)),
402  this, SLOT(webViewIconChanged()));
403  connect(webView, SIGNAL(titleChanged(const QString &)),
404  this, SLOT(webViewTitleChanged(const QString &)));
405  connect(webView, SIGNAL(urlChanged(const QUrl &)),
406  this, SLOT(webViewUrlChanged(const QUrl &)));
407  connect(webView, SIGNAL(search(const QUrl&, TabWidget::OpenUrlIn)),
408  this, SLOT(loadUrl(const QUrl&, TabWidget::OpenUrlIn)));
409  connect(webView->page(), SIGNAL(windowCloseRequested()),
410  this, SLOT(windowCloseRequested()));
411  connect(webView->page(), SIGNAL(printRequested(Q)),
412  this, SIGNAL(printRequested(webView->page())));
413  connect(webView->page(), SIGNAL(geometryChangeRequested(const QRect &)),
414  this, SLOT(geometryChangeRequestedCheck(const QRect &)));
415  connect(webView, &WebView::devToolsRequested, this, &TabWidget::devToolsRequested);
416 
417  WebViewWithSearch *webViewWithSearch = new WebViewWithSearch(webView, this);
418  addTab(webViewWithSearch, tr("Untitled"));
419  if (makeCurrent)
420  setCurrentWidget(webViewWithSearch);
421 
422  // webview actions
423  for (int i = 0; i < m_actions.count(); ++i) {
424  WebActionMapper *mapper = m_actions[i];
425  mapper->addChild(webView->page()->action(mapper->webAction()));
426  }
427 
428  if (count() == 1)
429  currentChanged(currentIndex());
430  emit tabsChanged();
431  return webView;
432 }

449 void TabWidget::bookmarkTabs()
450 {

452  dialog.setFolder(true);
453  dialog.setTitle(tr("Saved Tabs"));
454  dialog.exec();
455

458  return;

467  BookmarkNode *bookmark = new BookmarkNode(BookmarkNode::Bookmark);
468  bookmark->url = url;
469  bookmark->title = title;
470  BrowserApplication::bookmarksManager()->addBookmark(folder, bookmark);
471  }
472 }

777 QUrl TabWidget::guessUrlFromString(const QString &string)
778 {
779  OpenSearchManager *manager = ToolbarSearch::openSearchManager();
780  QUrl url = manager->convertKeywordSearchToUrl(string);
781  if (url.isValid())
782  return url;
783 
784  url = QUrl::fromUserInput(string);
785 
786  if (url.scheme() == QLatin1String("about")
787  && url.path() == QLatin1String("home"))
788  url = QUrl(QLatin1String("https://endorphinbrowser.github.io/newTab/0.13.0-preview1/"));
789 
790  // QUrl::isValid() is too much tolerant.
791  // We actually want to check if the url conforms to the RFC, which QUrl::isValid() doesn't state.
792  if (!url.scheme().isEmpty() && (!url.host().isEmpty() || !url.path().isEmpty()))
793  return url;
794 
795  QSettings settings;
796  settings.beginGroup(QLatin1String("urlloading"));
797  bool search = settings.value(QLatin1String("searchEngineFallback"), false).toBool();
798 
799  if (search) {
800  url = ToolbarSearch::openSearchManager()->currentEngine()->searchUrl(string.trimmed());
801  } else {
802  QString urlString = QLatin1String("http://") + string.trimmed();
803  url = QUrl::fromEncoded(urlString.toUtf8(), QUrl::TolerantMode);
804  }
805 
806  return url;
807 }
