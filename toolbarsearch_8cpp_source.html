93 ToolbarSearch::ToolbarSearch(QWidget *parent)
94  : SearchLineEdit(parent)
95  , m_suggestionsEnabled(true)
96  , m_autosaver(new AutoSaver(this))
97  , m_maxSavedSearches(10)
98  , m_model(new QStandardItemModel(this))
99  , m_suggestionsItem(nullptr)
100  , m_recentSearchesItem(nullptr)
101  , m_suggestTimer(nullptr)
102  , m_completer(nullptr)
103 {
104  connect(openSearchManager(), SIGNAL(currentEngineChanged()),
105  this, SLOT(currentEngineChanged()));
106 
107  m_completer = new QCompleter(m_model, this);
108  m_completer->setCompletionMode(QCompleter::UnfilteredPopupCompletion);
109  setCompleter(m_completer);
110 
111  searchButton()->setShowMenuTriangle(true);
112 
113  connect(searchButton(), SIGNAL(clicked()),
114  this, SLOT(showEnginesMenu()));
115  connect(this, SIGNAL(returnPressed()),
116  this, SLOT(searchNow()));
117 
118  load();
119 
120  currentEngineChanged();
121 }

130 void ToolbarSearch::currentEngineChanged()
131 {
132  OpenSearchEngine *newEngine = openSearchManager()->currentEngine();
133  Q_ASSERT(newEngine);
134  if (!newEngine)
135  return;
136 
137  if (m_suggestionsEnabled) {
138  if (openSearchManager()->engineExists(m_currentEngine)) {
139  OpenSearchEngine *oldEngine = openSearchManager()->engine(m_currentEngine);
140  disconnect(oldEngine, SIGNAL(suggestions(const QStringList &)),
141  this, SLOT(newSuggestions(const QStringList &)));
142  }
143 
144  connect(newEngine, SIGNAL(suggestions(const QStringList &)),
145  this, SLOT(newSuggestions(const QStringList &)));
146  }
147 
148  setInactiveText(newEngine->name());
149  m_currentEngine = newEngine->name();
150  m_suggestions.clear();
151  setupList();
152 }

160 bool ToolbarSearch::completerHighlighted(const QModelIndex &index)
161 {
162  if (m_suggestionsItem && m_suggestionsItem->index().row() == index.row())
163  return false;
164  if (m_recentSearchesItem && m_recentSearchesItem->index().row() == index.row())
165  return false;
166  setText(index.data().toString());
167  return true;
168 }

247 void ToolbarSearch::searchNow()
248 {
249  OpenSearchEngine *engine = openSearchManager()->currentEngine();
250  Q_ASSERT(engine);
251  if (!engine)
252  return;
253 
254  QString searchText = text();
255 
256 // QWebEngineSettings *globalSettings = QWebEngineSettings::globalSettings();
257 // if (!globalSettings->testAttribute(QWebEngineSettings::PrivateBrowsingEnabled)) {
258  QStringList newList = m_recentSearches;
259  if (newList.contains(searchText))
260  newList.removeAt(newList.indexOf(searchText));
261  newList.prepend(searchText);
262  if (newList.size() >= m_maxSavedSearches)
263  newList.removeLast();
264 
265  m_recentSearches = newList;
266  m_autosaver->changeOccurred();
267 // }
268 
269  QUrl searchUrl = engine->searchUrl(searchText);
270  TabWidget::OpenUrlIn tab = TabWidget::CurrentTab;
271  if (qApp->keyboardModifiers() == Qt::AltModifier)
272  tab = TabWidget::NewSelectedTab;
273  emit search(searchUrl, tab);
274 }

377 void ToolbarSearch::setupList()
378 {
379  if (m_suggestions.isEmpty()
380  || (m_model->rowCount() > 0
381  && m_model->item(0) != m_suggestionsItem)) {
382  m_model->clear();
383  m_suggestionsItem = nullptr;
384  } else {
385  m_model->removeRows(1, m_model->rowCount() - 1);
386  }
387 
388  QFont lightFont;
389  lightFont.setWeight(QFont::Light);
390  if (!m_suggestions.isEmpty()) {
391  if (m_model->rowCount() == 0) {
392  if (!m_suggestionsItem) {
393  m_suggestionsItem = new QStandardItem();
394  m_suggestionsItem->setFont(lightFont);
395  retranslate();
396  }
397  m_model->appendRow(m_suggestionsItem);
398  }
399  for (int i = 0; i < m_suggestions.count(); ++i) {
400  const QString &text = m_suggestions.at(i);
401  m_model->appendRow(new QStandardItem(text));
402  }
403  }
404 
405  if (m_recentSearches.isEmpty()) {
406  m_recentSearchesItem = new QStandardItem(tr("No Recent Searches"));
407  m_recentSearchesItem->setFont(lightFont);
408  m_model->appendRow(m_recentSearchesItem);
409  } else {
410  m_recentSearchesItem = new QStandardItem(tr("Recent Searches"));
411  m_recentSearchesItem->setFont(lightFont);
412  m_model->appendRow(m_recentSearchesItem);
413  for (int i = 0; i < m_recentSearches.count(); ++i) {
414  QString text = m_recentSearches.at(i);
415  m_model->appendRow(new QStandardItem(text));
416  }
417  }
418 
419  QAbstractItemView *view = completer()->popup();
420  view->setFixedHeight(view->sizeHintForRow(0) * m_model->rowCount() + view->frameWidth() * 2);
421 }
