82 void UserAgentMenu::addActionsFromFile(const QString &fileName)
83 {
84  QFile file(fileName);
85  if (!file.open(QFile::ReadOnly))
86  return;
87 
88  QString currentUserAgentString = WebPage::userAgent();
89  QXmlStreamReader xml(&file);
90  while (!xml.atEnd()) {
91  xml.readNext();
92  if (xml.isStartElement() && xml.name() == QLatin1String("separator")) {
93  addSeparator();
94  continue;
95  }
96  if (xml.isStartElement() && xml.name() == QLatin1String("useragent")) {
97  QXmlStreamAttributes attributes = xml.attributes();
98  QString title = attributes.value(QLatin1String("description")).toString();
99  QString userAgent = attributes.value(QLatin1String("useragent")).toString();
100 
101  QAction *action = new QAction(this);
102  action->setText(title);
103  action->setData(userAgent);
104  action->setToolTip(userAgent);
105  action->setCheckable(true);
106  action->setChecked(userAgent == currentUserAgentString);
107  connect(action, SIGNAL(triggered()), this, SLOT(changeUserAgent()));
108  addAction(action);
109  }
110  }
111  if (xml.hasError()) {
112  qDebug() << "Error reading custom user agents" << xml.errorString();
113  // ... do error handling
114  }
115 }
