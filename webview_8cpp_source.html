498 int WebView::levelForZoom(int zoom)
499 {
500  int i;
501 
502  i = m_zoomLevels.indexOf(zoom);
503  if (i >= 0)
504  return i;
505 
506  for (i = 0 ; i < m_zoomLevels.count(); ++i)
507  if (zoom <= m_zoomLevels[i])
508  break;
509 
510  if (i == m_zoomLevels.count())
511  return i - 1;
512  if (i == 0)
513  return i;
514 
515  if (zoom - m_zoomLevels[i-1] > m_zoomLevels[i] - zoom)
516  return i;
517  else
518  return i-1;
519 }

559 void WebView::loadUrl(const QUrl &url, const QString &title)
560 {
561  if (url.scheme() == QLatin1String("javascript")) {
562  QString scriptSource = QUrl::fromPercentEncoding(url.toString(Q_FLAGS(QUrl::TolerantMode|QUrl::RemoveScheme)).toLatin1());
563  QEventLoop loop;
564  QObject::connect(this, SIGNAL(notifyRanJavaScript()), &loop, SLOT(quit()));
565  page()->runJavaScript(scriptSource, [this](const QVariant &v)
566  {
567  this->ranJavaScript();
568  });
569  loop.exec();
570  return;
571  }
572  m_initialUrl = url;
573  if (!title.isEmpty())
574  emit titleChanged(tr("Loading..."));
575  else
576  emit titleChanged(title);
577  QUrl oldurl = QUrl(url.toString());
578  QUrl newurl = QUrl(url.toString());
579  if(newurl.toString().startsWith(QLatin1String("endorphin://"))) {
580  QString url_tmp = newurl.toString().mid(12);
581  newurl = QUrl(QLatin1String("qrc:/") + url_tmp);
582  QString urlstr = newurl.toString();
583  if(!urlstr.endsWith(QLatin1String(".html"))) {
584  newurl = QUrl(urlstr + QLatin1String(".html"));
585  }
586  }
587  load(newurl);
588 }
